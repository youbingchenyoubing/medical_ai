### 1.1 特征提取完整流程

#### PyRadiomics实战代码

```python
import os
import pandas as pd
import SimpleITK as sitk
from radiomics import featureextractor
import numpy as np
from tqdm import tqdm

class RadiomicsFeatureExtractor:
    def __init__(self, params_yaml=None):
        """
        初始化特征提取器
        
        参数:
            params_yaml: 参数配置文件路径
        """
        self.extractor = featureextractor.RadiomicsFeatureExtractor(params_yaml)
        
        # 设置常用参数
        self.extractor.settings['binWidth'] = 25  # 灰度bin宽度
        self.extractor.settings['resampledPixelSpacing'] = [1, 1, 1]  # 重采样体素大小
        self.extractor.settings['interpolator'] = 'sitkBSpline'  # 插值方法
        self.extractor.settings['normalize'] = True  # 是否归一化
        self.extractor.settings['force2D'] = False  # 是否强制2D
        self.extractor.settings['textureSampleDistance'] = 1  # 纹理采样距离
        
    def extract_features_from_single_case(self, image_path, mask_path):
        """
        从单个病例提取特征
        
        参数:
            image_path: 图像文件路径
            mask_path: mask文件路径
            
        返回:
            features: 特征字典
        """
        try:
            # 读取图像和mask
            image = sitk.ReadImage(image_path)
            mask = sitk.ReadImage(mask_path)
            
            # 提取特征
            features = self.extractor.execute(image, mask)
            
            return features
            
        except Exception as e:
            print(f"Error processing {image_path}: {str(e)}")
            return None
    
    def batch_extract_features(self, image_dir, mask_dir, output_csv):
        """
        批量提取特征
        
        参数:
            image_dir: 图像目录
            mask_dir: mask目录
            output_csv: 输出CSV文件路径
        """
        all_features = []
        
        # 获取所有图像文件
        image_files = [f for f in os.listdir(image_dir) if f.endswith('.nii.gz')]
        
        for image_file in tqdm(image_files, desc="Extracting features"):
            case_id = image_file.replace('.nii.gz', '')
            image_path = os.path.join(image_dir, image_file)
            mask_path = os.path.join(mask_dir, f"{case_id}_mask.nii.gz")
            
            if not os.path.exists(mask_path):
                print(f"Mask not found for {case_id}")
                continue
                
            features = self.extract_features_from_single_case(image_path, mask_path)
            
            if features:
                features['case_id'] = case_id
                all_features.append(features)
        
        # 转换为DataFrame
        df = pd.DataFrame(all_features)
        
        # 保存到CSV
        df.to_csv(output_csv, index=False)
        print(f"Features saved to {output_csv}")
        
        return df

# 使用示例
if __name__ == "__main__":
    extractor = RadiomicsFeatureExtractor()
    df = extractor.batch_extract_features(
        image_dir='data/images',
        mask_dir='data/masks',
        output_csv='results/features.csv'
    )
```

### 1.2 特征筛选算法实现

#### LASSO特征选择

```python
import numpy as np
import pandas as pd
from sklearn.linear_model import LassoCV
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import cross_val_score
import matplotlib.pyplot as plt

class LassoFeatureSelector:
    def __init__(self, cv_folds=5, random_state=42):
        """
        LASSO特征选择器
        
        参数:
            cv_folds: 交叉验证折数
            random_state: 随机种子
        """
        self.cv_folds = cv_folds
        self.random_state = random_state
        self.scaler = StandardScaler()
        self.lasso = None
        self.selected_features = None
        self.coef_ = None
        
    def fit(self, X, y):
        """
        训练LASSO模型并选择特征
        
        参数:
            X: 特征矩阵 (n_samples, n_features)
            y: 标签 (n_samples,)
        """
        # 标准化
        X_scaled = self.scaler.fit_transform(X)
        
        # LASSO回归，自动选择最优alpha
        self.lasso = LassoCV(
            cv=self.cv_folds,
            random_state=self.random_state,
            max_iter=10000,
            n_jobs=-1
        )
        self.lasso.fit(X_scaled, y)
        
        # 获取非零系数的特征
        self.coef_ = self.lasso.coef_
        self.selected_features = np.where(self.coef_ != 0)[0]
        
        print(f"Optimal alpha: {self.lasso.alpha_:.6f}")
        print(f"Selected {len(self.selected_features)} features out of {X.shape[1]}")
        
        return self
    
    def transform(self, X):
        """
        使用选中的特征转换数据
        
        参数:
            X: 特征矩阵
            
        返回:
            X_selected: 选中特征组成的矩阵
        """
        return X[:, self.selected_features]
    
    def fit_transform(self, X, y):
        """
        拟合并转换
        """
        self.fit(X, y)
        return self.transform(X)
    
    def plot_coefficients(self, feature_names, save_path=None):
        """
        绘制特征系数图
        """
        # 只绘制非零系数
        non_zero_idx = np.where(self.coef_ != 0)[0]
        non_zero_coef = self.coef_[non_zero_idx]
        non_zero_names = [feature_names[i] for i in non_zero_idx]
        
        # 排序
        sorted_idx = np.argsort(np.abs(non_zero_coef))[::-1]
        
        plt.figure(figsize=(12, 8))
        plt.barh(range(len(sorted_idx)), non_zero_coef[sorted_idx])
        plt.yticks(range(len(sorted_idx)), [non_zero_names[i] for i in sorted_idx])
        plt.xlabel('Coefficient')
        plt.title('LASSO Selected Features')
        plt.tight_layout()
        
        if save_path:
            plt.savefig(save_path, dpi=300, bbox_inches='tight')
        plt.show()
    
    def get_selected_feature_names(self, feature_names):
        """
        获取选中特征的名称
        """
        return [feature_names[i] for i in self.selected_features]

# 使用示例
if __name__ == "__main__":
    # 加载数据
    df = pd.read_csv('features.csv')
    X = df.drop(['case_id', 'label'], axis=1).values
    y = df['label'].values
```

#### 递归特征消除（RFE）

```python
from sklearn.feature_selection import RFE
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import cross_val_score
import numpy as np

class RFEFeatureSelector:
    def __init__(self, n_features_to_select=10, estimator_type='lr'):
        """
        RFE特征选择器
        
        参数:
            n_features_to_select: 要选择的特征数量
            estimator_type: 基础估计器类型 ('lr' 或 'rf')
        """
        self.n_features_to_select = n_features_to_select
        
        if estimator_type == 'lr':
            self.estimator = LogisticRegression(
                max_iter=1000,
                random_state=42,
                class_weight='balanced'
            )
        elif estimator_type == 'rf':
            self.estimator = RandomForestClassifier(
                n_estimators=100,
                random_state=42,
                n_jobs=-1
            )
        
        self.rfe = None
        self.selected_features = None
        
    def fit(self, X, y):
        """
        训练RFE模型
        """
        self.rfe = RFE(
            self.estimator,
            n_features_to_select=self.n_features_to_select,
            step=1
        )
        self.rfe.fit(X, y)
        
        self.selected_features = np.where(self.rfe.support_)[0]
        
        print(f"Selected {len(self.selected_features)} features")
        print(f"Ranking: {self.rfe.ranking_}")
        
        return self
    
    def transform(self, X):
        """
        转换数据
        """
        return self.rfe.transform(X)
    
    def fit_transform(self, X, y):
        """
        拟合并转换
        """
        self.fit(X, y)
        return self.transform(X)
    
    def get_selected_feature_names(self, feature_names):
        """
        获取选中特征名称
        """
        return [feature_names[i] for i in self.selected_features]

# 使用示例
selector = RFEFeatureSelector(n_features_to_select=10, estimator_type='lr')
X_selected = selector.fit_transform(X, y)
selected_names = selector.get_selected_feature_names(feature_names)
```

### 1.3 模型构建与评估

#### 完整的模型训练流程

```python
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split, cross_val_score, StratifiedKFold
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression
from sklearn.svm import SVC
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.metrics import (
    roc_auc_score, accuracy_score, sensitivity_score, specificity_score,
    confusion_matrix, classification_report, roc_curve, auc
)
import matplotlib.pyplot as plt
import seaborn as sns

class RadiomicsModelTrainer:
    def __init__(self, random_state=42):
        """
        影像组学模型训练器
        """
        self.random_state = random_state
        self.scaler = StandardScaler()
        self.models = {}
        self.results = {}
        
    def prepare_data(self, X, y, test_size=0.2):
        """
        准备训练和测试数据
        
        参数:
            X: 特征矩阵
            y: 标签
            test_size: 测试集比例
        """
        # 分割数据
        self.X_train, self.X_test, self.y_train, self.y_test = train_test_split(
            X, y, test_size=test_size, random_state=self.random_state, stratify=y
        )
        
        # 标准化
        self.X_train_scaled = self.scaler.fit_transform(self.X_train)
        self.X_test_scaled = self.scaler.transform(self.X_test)
        
        print(f"Training set: {self.X_train.shape[0]} samples")
        print(f"Test set: {self.X_test.shape[0]} samples")
        print(f"Class distribution (train): {np.bincount(self.y_train)}")
        print(f"Class distribution (test): {np.bincount(self.y_test)}")
        
        return self
    
    def train_models(self, model_types=['lr', 'svm', 'rf', 'gb']):
        """
        训练多个模型
        
        参数:
            model_types: 要训练的模型类型列表
        """
        for model_type in model_types:
            print(f"\n{'='*50}")
            print(f"Training {model_type.upper()} model...")
            print(f"{'='*50}")
            
            if model_type == 'lr':
                model = LogisticRegression(
                    max_iter=1000,
                    random_state=self.random_state,
                    class_weight='balanced'
                )
            elif model_type == 'svm':
                model = SVC(
                    kernel='rbf',
                    probability=True,
                    random_state=self.random_state,
                    class_weight='balanced'
                )
            elif model_type == 'rf':
                model = RandomForestClassifier(
                    n_estimators=500,
                    max_depth=10,
                    random_state=self.random_state,
                    class_weight='balanced',
                    n_jobs=-1
                )
            elif model_type == 'gb':
                model = GradientBoostingClassifier(
                    n_estimators=500,
                    max_depth=5,
                    learning_rate=0.01,
                    random_state=self.random_state
                )
            
            # 交叉验证
            cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=self.random_state)
            cv_scores = cross_val_score(model, self.X_train_scaled, self.y_train, 
                                       cv=cv, scoring='roc_auc', n_jobs=-1)
            
            print(f"CV AUC: {cv_scores.mean():.3f} ± {cv_scores.std():.3f}")
            
            # 训练模型
            model.fit(self.X_train_scaled, self.y_train)
            
            # 预测
            y_pred = model.predict(self.X_test_scaled)
            y_prob = model.predict_proba(self.X_test_scaled)[:, 1]
            
            # 评估
            metrics = self._evaluate_model(self.y_test, y_pred, y_prob)
            
            # 保存结果
            self.models[model_type] = model
            self.results[model_type] = {
                'cv_scores': cv_scores,
                'metrics': metrics,
                'y_pred': y_pred,
                'y_prob': y_prob
            }
            
            print(f"\nTest set performance:")
            print(f"AUC: {metrics['auc']:.3f}")
            print(f"Accuracy: {metrics['accuracy']:.3f}")
            print(f"Sensitivity: {metrics['sensitivity']:.3f}")
            print(f"Specificity: {metrics['specificity']:.3f}")
        
        return self
    
    def _evaluate_model(self, y_true, y_pred, y_prob):
        """
        评估模型性能
        """
        # 计算指标
        auc = roc_auc_score(y_true, y_prob)
        accuracy = accuracy_score(y_true, y_pred)
        
        # 混淆矩阵
        tn, fp, fn, tp = confusion_matrix(y_true, y_pred).ravel()
        sensitivity = tp / (tp + fn)
        specificity = tn / (tn + fp)
        
        return {
            'auc': auc,
            'accuracy': accuracy,
            'sensitivity': sensitivity,
            'specificity': specificity,
            'confusion_matrix': confusion_matrix(y_true, y_pred)
        }
    
    def plot_roc_curves(self, save_path=None):
        """
        绘制所有模型的ROC曲线
        """
        plt.figure(figsize=(10, 8))
        
        for model_type, result in self.results.items():
            fpr, tpr, _ = roc_curve(self.y_test, result['y_prob'])
            roc_auc = auc(fpr, tpr)
            plt.plot(fpr, tpr, lw=2, 
                    label=f'{model_type.upper()} (AUC = {roc_auc:.3f})')
        
        plt.plot([0, 1], [0, 1], 'k--', lw=2)
        plt.xlim([0.0, 1.0])
        plt.ylim([0.0, 1.05])
        plt.xlabel('False Positive Rate', fontsize=12)
        plt.ylabel('True Positive Rate', fontsize=12)
        plt.title('ROC Curves', fontsize=14)
        plt.legend(loc="lower right", fontsize=10)
        plt.grid(True, alpha=0.3)
        
        if save_path:
            plt.savefig(save_path, dpi=300, bbox_inches='tight')
        plt.show()
    
    def plot_confusion_matrices(self, save_path=None):
        """
        绘制混淆矩阵
        """
        n_models = len(self.models)
        fig, axes = plt.subplots(1, n_models, figsize=(5*n_models, 4))
        
        if n_models == 1:
            axes = [axes]
        
        for idx, (model_type, result) in enumerate(self.results.items()):
            cm = result['metrics']['confusion_matrix']
            sns.heatmap(cm, annot=True, fmt='d', cmap='Blues', ax=axes[idx])
            axes[idx].set_title(f'{model_type.upper()}', fontsize=12)
            axes[idx].set_xlabel('Predicted', fontsize=10)
            axes[idx].set_ylabel('Actual', fontsize=10)
        
        plt.tight_layout()
        
        if save_path:
            plt.savefig(save_path, dpi=300, bbox_inches='tight')
        plt.show()
    
    def get_best_model(self, metric='auc'):
        """
        获取最佳模型
        
        参数:
            metric: 评估指标 ('auc', 'accuracy', 'sensitivity', 'specificity')
        """
        best_score = 0
        best_model_type = None
        
        for model_type, result in self.results.items():
            score = result['metrics'][metric]
            if score > best_score:
                best_score = score
                best_model_type = model_type
        
        print(f"Best model: {best_model_type.upper()}")
        print(f"{metric.upper()}: {best_score:.3f}")
        
        return self.models[best_model_type], best_model_type

# 使用示例
if __name__ == "__main__":
    # 加载数据
    df = pd.read_csv('features_selected.csv')
    X = df.drop(['case_id', 'label'], axis=1).values
    y = df['label'].values
    
    # 训练模型
    trainer = RadiomicsModelTrainer(random_state=42)
    trainer.prepare_data(X, y, test_size=0.2)
    trainer.train_models(model_types=['lr', 'svm', 'rf', 'gb'])
    
    # 绘制结果
    trainer.plot_roc_curves('roc_curves.png')
    trainer.plot_confusion_matrices('confusion_matrices.png')
    
    # 获取最佳模型
    best_model, best_model_type = trainer.get_best_model(metric='auc')
```

```python
import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LassoCV, LogisticRegression
from sklearn.svm import SVC
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import roc_auc_score, accuracy_score
from radiomics import featureextractor
import SimpleITK as sitk

class NSCLC_Radiomics_Pipeline:
    """
    NSCLC影像组学完整流程
    """
    def __init__(self):
        self.extractor = featureextractor.RadiomicsFeatureExtractor()
        self.scaler = StandardScaler()
        self.lasso = None
        self.model = None
        
    def step1_feature_extraction(self, image_dir, mask_dir, output_csv):
        """
        步骤1: 特征提取
        """
        print("Step 1: Extracting radiomics features...")
        
        # 设置参数
        self.extractor.settings['binWidth'] = 25
        self.extractor.settings['resampledPixelSpacing'] = [1, 1, 1]
        
        # 提取特征（代码见1.1节）
        # ...
        
        return self
    
    def step2_feature_selection(self, X, y, n_features=10):
        """
        步骤2: 特征筛选
        """
        print(f"Step 2: Selecting top {n_features} features...")
        
        # 标准化
        X_scaled = self.scaler.fit_transform(X)
        
        # LASSO特征选择
        self.lasso = LassoCV(cv=5, random_state=42, max_iter=10000)
        self.lasso.fit(X_scaled, y)
        
        # 选择非零系数的特征
        coef = self.lasso.coef_
        selected_idx = np.where(coef != 0)[0]
        
        # 如果特征太多，选择系数绝对值最大的n_features个
        if len(selected_idx) > n_features:
            top_idx = np.argsort(np.abs(coef))[::-1][:n_features]
            selected_idx = top_idx
        
        print(f"Selected {len(selected_idx)} features")
        
        return X_scaled[:, selected_idx], selected_idx
    
    def step3_model_training(self, X_train, y_train, X_val, y_val):
        """
        步骤3: 模型训练
        """
        print("Step 3: Training models...")
        
        # 尝试多个模型
        models = {
            'LR': LogisticRegression(max_iter=1000, random_state=42),
            'SVM': SVC(kernel='rbf', probability=True, random_state=42),
            'RF': RandomForestClassifier(n_estimators=500, random_state=42)
        }
        
        best_model = None
        best_auc = 0
        best_name = None
        
        for name, model in models.items():
            # 训练
            model.fit(X_train, y_train)
            
            # 验证
            y_prob = model.predict_proba(X_val)[:, 1]
            auc = roc_auc_score(y_val, y_prob)
            
            print(f"{name}: AUC = {auc:.3f}")
            
            if auc > best_auc:
                best_auc = auc
                best_model = model
                best_name = name
        
        print(f"\nBest model: {best_name} (AUC = {best_auc:.3f})")
        self.model = best_model
        
        return self
    
    def step4_external_validation(self, X_test, y_test):
        """
        步骤4: 外部验证
        """
        print("Step 4: External validation...")
        
        y_prob = self.model.predict_proba(X_test)[:, 1]
        y_pred = self.model.predict(X_test)
        
        auc = roc_auc_score(y_test, y_prob)
        acc = accuracy_score(y_test, y_pred)
        
        print(f"External validation AUC: {auc:.3f}")
        print(f"External validation Accuracy: {acc:.3f}")
        
        return auc, acc
```

```python
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.svm import SVC
from sklearn.metrics import roc_auc_score, accuracy_score, confusion_matrix
import torch
import torch.nn as nn
from torchvision import models, transforms
from PIL import Image

class HCC_Immunotherapy_Predictor:
    """
    HCC免疫治疗反应预测
    """
    def __init__(self):
        self.radiomics_extractor = None
        self.deep_model = None
        self.scaler = StandardScaler()
        
    def extract_radiomics_features(self, image_path, mask_path):
        """
        提取传统影像组学特征
        """
        from radiomics import featureextractor
        
        extractor = featureextractor.RadiomicsFeatureExtractor()
        extractor.settings['binWidth'] = 25
        extractor.settings['resampledPixelSpacing'] = [1, 1, 1]
        
        image = sitk.ReadImage(image_path)
        mask = sitk.ReadImage(mask_path)
        
        features = extractor.execute(image, mask)
        
        return features
    
    def extract_deep_features(self, image_3d):
        """
        提取深度学习特征
        
        参数:
            image_3d: 3D numpy数组
        """
        # 加载预训练ResNet
        model = models.resnet50(pretrained=True)
        model = nn.Sequential(*list(model.children())[:-1])
        model.eval()
        
        # 预处理
        preprocess = transforms.Compose([
            transforms.Resize(224),
            transforms.CenterCrop(224),
            transforms.ToTensor(),
            transforms.Normalize(mean=[0.485, 0.456, 0.406],
                               std=[0.229, 0.224, 0.225]),
        ])
        
        # 提取中间层特征
        deep_features = []
        
        # 对每个切片提取特征
        for slice_idx in range(image_3d.shape[2]):
            slice_2d = image_3d[:, :, slice_idx]
            
            # 转换为PIL图像并复制为3通道
            slice_pil = Image.fromarray(slice_2d).convert('RGB')
            slice_tensor = preprocess(slice_pil).unsqueeze(0)
            
            with torch.no_grad():
                features = model(slice_tensor)
            
            deep_features.append(features.squeeze().numpy())
        
        # 平均所有切片的特征
        deep_features = np.mean(deep_features, axis=0)
        
        return deep_features
    
    def build_fusion_model(self, radiomics_features, deep_features, clinical_features=None):
        """
        构建融合模型
        
        参数:
            radiomics_features: 影像组学特征
            deep_features: 深度学习特征
            clinical_features: 临床特征（可选）
        """
        # 标准化
        radiomics_scaled = self.scaler.fit_transform(radiomics_features)
        
        # 特征级融合
        if clinical_features is not None:
            fused_features = np.concatenate([
                radiomics_scaled,
                deep_features,
                clinical_features
            ], axis=1)
        else:
            fused_features = np.concatenate([
                radiomics_scaled,
                deep_features
            ], axis=1)
        
        return fused_features
    
    def train_model(self, X_train, y_train, X_val, y_val):
        """
        训练模型
        """
        # 尝试多个模型
        models = {
            'LR': LogisticRegression(max_iter=1000, random_state=42),
            'SVM': SVC(kernel='rbf', probability=True, random_state=42),
            'RF': RandomForestClassifier(n_estimators=500, random_state=42),
            'GB': GradientBoostingClassifier(n_estimators=500, random_state=42)
        }
        
        best_model = None
        best_auc = 0
        
        for name, model in models.items():
            model.fit(X_train, y_train)
            y_prob = model.predict_proba(X_val)[:, 1]
            auc = roc_auc_score(y_val, y_prob)
            
            print(f"{name}: AUC = {auc:.3f}")
            
            if auc > best_auc:
                best_auc = auc
                best_model = model
        
        self.model = best_model
        return best_model
```

```python
import numpy as np
from scipy import stats

def calculate_sample_size(effect_size=0.5, alpha=0.05, power=0.8):
    """
    计算所需样本量
    
    参数:
        effect_size: 效应量 (Cohen's d)
        alpha: 显著性水平
        power: 统计功效
        
    返回:
        n: 每组所需样本量
    """
    from scipy.stats import norm
    
    # 计算z值
    z_alpha = norm.ppf(1 - alpha/2)
    z_beta = norm.ppf(power)
    
    # 计算样本量
    n = 2 * ((z_alpha + z_beta) / effect_size) ** 2
    
    print(f"Effect size: {effect_size}")
    print(f"Alpha: {alpha}")
    print(f"Power: {power}")
    print(f"Required sample size per group: {int(np.ceil(n))}")
    
    return int(np.ceil(n))
```

### 3.3 特征稳定性评估

```python
from sklearn.metrics import intraclass_correlation_coefficient
import numpy as np

def calculate_icc(features_1, features_2):
    """
    计算组内相关系数（ICC）
    
    参数:
        features_1: 第一次提取的特征
        features_2: 第二次提取的特征
        
    返回:
        icc: ICC值
    """
    # 计算ICC
    n = len(features_1)
    mean_1 = np.mean(features_1)
    mean_2 = np.mean(features_2)
    
    # 组间方差
    ms_between = n * np.var([mean_1, mean_2])
    
    # 组内方差
    ms_within = np.mean([np.var(features_1), np.var(features_2)])
    
    # ICC
    icc = (ms_between - ms_within) / (ms_between + ms_within)
    
    return icc

def assess_feature_stability(feature_matrix_1, feature_matrix_2, threshold=0.75):
    """
    评估特征稳定性
    
    参数:
        feature_matrix_1: 第一次提取的特征矩阵 (n_samples, n_features)
        feature_matrix_2: 第二次提取的特征矩阵 (n_samples, n_features)
        threshold: ICC阈值
        
    返回:
        stable_features: 稳定特征的索引
    """
    n_features = feature_matrix_1.shape[1]
    icc_values = []
    
    for i in range(n_features):
        icc = calculate_icc(feature_matrix_1[:, i], feature_matrix_2[:, i])
        icc_values.append(icc)
    
    icc_values = np.array(icc_values)
    stable_features = np.where(icc_values >= threshold)[0]
    
    print(f"Features with ICC >= {threshold}: {len(stable_features)}/{n_features}")
    print(f"Mean ICC: {np.mean(icc_values):.3f}")
    
    return stable_features, icc_values
```

```python
import SimpleITK as sitk
import numpy as np

def preprocess_ct_image(image_path, target_spacing=[1, 1, 1]):
    """
    CT图像预处理
    
    参数:
        image_path: 图像路径
        target_spacing: 目标体素间距
        
    返回:
        preprocessed_image: 预处理后的图像
    """
    # 读取图像
    image = sitk.ReadImage(image_path)
    
    # 重采样
    original_spacing = image.GetSpacing()
    original_size = image.GetSize()
    
    new_size = [
        int(round(original_size[i] * (original_spacing[i] / target_spacing[i])))
        for i in range(3)
    ]
    
    resampler = sitk.ResampleImageFilter()
    resampler.SetOutputSpacing(target_spacing)
    resampler.SetSize(new_size)
    resampler.SetInterpolator(sitk.sitkBSpline)
    resampler.SetOutputOrigin(image.GetOrigin())
    resampler.SetOutputDirection(image.GetDirection())
    
    resampled_image = resampler.Execute(image)
    
    # 窗宽窗位调整（肺窗）
    window_level = -600
    window_width = 1500
    
    min_val = window_level - window_width / 2
    max_val = window_level + window_width / 2
    
    windowed_image = sitk.Clamp(resampled_image, 
                                lowerBound=min_val, 
                                upperBound=max_val)
    
    return windowed_image
```

```python
def segment_tumor_automatically(image, seed_point=None):
    """
    自动肿瘤分割
    
    参数:
        image: SimpleITK图像
        seed_point: 种子点（可选）
        
    返回:
        mask: 分割mask
    """

    image_array = sitk.GetArrayFromImage(image)

    threshold = -150  # CT值阈值
    binary = image_array > threshold
    
    # 形态学操作
    binary = morphology.remove_small_objects(binary, min_size=100)
    binary = morphology.remove_small_holes(binary, area_threshold=100)
    
    # 如果提供了种子点，使用区域生长
    if seed_point:
        # 转换种子点坐标
        seed_index = image.TransformPhysicalPointToIndex(seed_point)
        
        # 区域生长
        seg = sitk.Image(image.GetSize(), sitk.sitkUInt8)
        seg.CopyInformation(image)
        seg[seed_index] = 1
        
        # 连通阈值分割
        seg = sitk.ConnectedThreshold(
            image, 
            seedList=[seed_index],
            lower=-150,
            upper=100
        )
        
        # 形态学操作
        seg = sitk.BinaryMorphologicalClosing(seg, [2, 2, 2])
        seg = sitk.BinaryMorphologicalOpening(seg, [1, 1, 1])
        
        return seg
    
    # 转换回SimpleITK图像
    mask = sitk.GetImageFromArray(binary.astype(np.uint8))
    mask.CopyInformation(image)
    
    return mask
```

```python
    # 按相同顺序排列
```

```python
        method: 方法 ('smote', 'undersample', 'combine')
    """
    print(f"Original class distribution: {np.bincount(y)}")
    
    if method == 'smote':
        sampler = SMOTE(random_state=42)
    elif method == 'undersample':
        sampler = RandomUnderSampler(random_state=42)
    elif method == 'combine':
        sampler = SMOTETomek(random_state=42)
    
    X_resampled, y_resampled = sampler.fit_resample(X, y)
    
    print(f"Resampled class distribution: {np.bincount(y_resampled)}")
    
    return X_resampled, y_resampled
```

```python
from sklearn.model_selection import learning_curve
import matplotlib.pyplot as plt

def plot_learning_curve(estimator, X, y, cv=5):
    """
    绘制学习曲线，诊断过拟合
    """
    train_sizes, train_scores, val_scores = learning_curve(
        estimator, X, y, cv=cv, n_jobs=-1,
        train_sizes=np.linspace(0.1, 1.0, 10),
        scoring='roc_auc'
    )
    
    train_scores_mean = np.mean(train_scores, axis=1)
    train_scores_std = np.std(train_scores, axis=1)
    val_scores_mean = np.mean(val_scores, axis=1)
    val_scores_std = np.std(val_scores, axis=1)
    
    plt.figure(figsize=(10, 6))
    plt.plot(train_sizes, train_scores_mean, 'o-', color='r', label='Training score')
    plt.fill_between(train_sizes, train_scores_mean - train_scores_std,
                     train_scores_mean + train_scores_std, alpha=0.1, color='r')
    plt.plot(train_sizes, val_scores_mean, 'o-', color='g', label='Cross-validation score')
    plt.fill_between(train_sizes, val_scores_mean - val_scores_std,
                     val_scores_mean + val_scores_std, alpha=0.1, color='g')
    
    plt.xlabel('Training examples')
    plt.ylabel('Score')
    plt.title('Learning Curve')
    plt.legend(loc='best')
    plt.grid(True)
    plt.show()

# 使用示例
from sklearn.linear_model import LogisticRegression
estimator = LogisticRegression(max_iter=1000)
plot_learning_curve(estimator, X, y)
```

```python
import shap
import matplotlib.pyplot as plt

def explain_model_with_shap(model, X, feature_names):
    """
    使用SHAP值解释模型
    """
    # 创建SHAP解释器
    explainer = shap.TreeExplainer(model) if hasattr(model, 'estimators_') else shap.LinearExplainer(model, X)
    
    # 计算SHAP值
    shap_values = explainer.shap_values(X)
    
    # 绘制摘要图
    plt.figure(figsize=(10, 8))
    shap.summary_plot(shap_values, X, feature_names=feature_names, show=False)
    plt.tight_layout()
    plt.savefig('shap_summary.png', dpi=300, bbox_inches='tight')
    plt.show()
    
    # 绘制单个样本的解释
    plt.figure(figsize=(10, 6))
    shap.force_plot(explainer.expected_value, shap_values[0], X[0], 
                   feature_names=feature_names, matplotlib=True, show=False)
    plt.tight_layout()
    plt.savefig('shap_force_plot.png', dpi=300, bbox_inches='tight')
    plt.show()
    
    return shap_values

# 使用示例（随机森林）
from sklearn.ensemble import RandomForestClassifier

model.fit(X_train, y_train)
```
