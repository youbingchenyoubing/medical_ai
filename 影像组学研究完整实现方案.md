## 1. 研究主题一：NSCLC免疫治疗反应预测

### 1.1 研究背景

**临床问题**：非小细胞肺癌（NSCLC）患者接受免疫检查点抑制剂（ICI）治疗后，仅20-30%患者有效，需要预测治疗反应的生物标志物。

**研究目标**：基于CT影像组学特征预测NSCLC患者免疫治疗反应（ORR：客观缓解率）。

### 1.2 最新研究方法（2024-2025）

**参考论文**：
- "CT-based radiomics for prediction of immunotherapy response in NSCLC" (Radiology, 2024)
- "Deep learning radiomics signature for predicting response to immune checkpoint inhibitors" (Nature Communications, 2024)

**核心方法**：
1. 传统影像组学特征提取
2. 深度学习特征提取（ResNet50）
3. 临床特征整合（年龄、性别、吸烟史、PD-L1表达）
4. 多模态融合模型

#### 数据预处理模块（src/data_preprocessing.py）

```python
import os
import SimpleITK as sitk
import numpy as np
import pandas as pd
import yaml
from pathlib import Path

class DataPreprocessor:
    def __init__(self, config_path="config.yaml"):
        with open(config_path, 'r') as f:
            self.config = yaml.safe_load(f)
        
        self.target_spacing = self.config['preprocessing']['target_spacing']
        self.window_level = self.config['preprocessing']['window_level']
        self.window_width = self.config['preprocessing']['window_width']
    
    def preprocess_ct_image(self, image_path, output_path=None):
        """
        CT图像预处理
        """
        # 读取DICOM系列
        reader = sitk.ImageSeriesReader()
        dicom_names = reader.GetGDCMSeriesFileNames(image_path)
        reader.SetFileNames(dicom_names)
        image = reader.Execute()
        
        # 重采样
        original_spacing = image.GetSpacing()
        original_size = image.GetSize()
        
        new_size = [
            int(round(original_size[i] * (original_spacing[i] / self.target_spacing[i])))
            for i in range(3)
        ]
        
        resampler = sitk.ResampleImageFilter()
        resampler.SetOutputSpacing(self.target_spacing)
        resampler.SetSize(new_size)
        resampler.SetInterpolator(sitk.sitkBSpline)
        resampler.SetOutputOrigin(image.GetOrigin())
        resampler.SetOutputDirection(image.GetDirection())
        
        resampled_image = resampler.Execute(image)
        
        # 窗宽窗位调整
        min_val = self.window_level - self.window_width / 2
        max_val = self.window_level + self.window_width / 2
        
        windowed_image = sitk.Clamp(resampled_image, 
                                    lowerBound=min_val, 
                                    upperBound=max_val)
        
        # 保存
        if output_path:
            sitk.WriteImage(windowed_image, output_path)
        
        return windowed_image
    
    def batch_preprocess(self, raw_dir, output_dir):
        """
        批量预处理
        """
        os.makedirs(output_dir, exist_ok=True)
        
        # 获取所有病例
        cases = [d for d in os.listdir(raw_dir) if os.path.isdir(os.path.join(raw_dir, d))]
        
        for case in cases:
            print(f"Processing {case}...")
            case_dir = os.path.join(raw_dir, case)
            output_path = os.path.join(output_dir, f"{case}.nii.gz")
            
            try:
                self.preprocess_ct_image(case_dir, output_path)
            except Exception as e:
                print(f"Error processing {case}: {str(e)}")
                continue
        
        print(f"Preprocessing completed. Processed {len(cases)} cases.")

# 使用示例
if __name__ == "__main__":
    preprocessor = DataPreprocessor()
    preprocessor.batch_preprocess(
        raw_dir="data/raw",
        output_dir="data/processed"
    )
```

#### 特征提取模块（src/feature_extraction.py）

```python
import os
import pandas as pd
import SimpleITK as sitk
from radiomics import featureextractor
import numpy as np
from tqdm import tqdm
import yaml

class FeatureExtractor:
    def __init__(self, config_path="config.yaml"):
        with open(config_path, 'r') as f:
            self.config = yaml.safe_load(f)
        
        # 初始化PyRadiomics
        self.extractor = featureextractor.RadiomicsFeatureExtractor()
        
        # 设置参数
        params = self.config['feature_extraction']
        self.extractor.settings['binWidth'] = params['bin_width']
        self.extractor.settings['resampledPixelSpacing'] = params['resampled_spacing']
        self.extractor.settings['interpolator'] = params['interpolator']
        self.extractor.settings['normalize'] = True
        
    def extract_features_single_case(self, image_path, mask_path):
        """
        提取单个病例的特征
        """
        try:
            # 读取图像和mask
            image = sitk.ReadImage(image_path)
            mask = sitk.ReadImage(mask_path)
            
            # 提取特征
            features = self.extractor.execute(image, mask)
            
            # 过滤掉元数据，只保留特征值
            feature_dict = {}
            for key, value in features.items():
                if key.startswith('original_'):
                    feature_dict[key] = value
            
            return feature_dict
            
        except Exception as e:
            print(f"Error extracting features: {str(e)}")
            return None
    
    def extract_features_batch(self, image_dir, mask_dir, output_csv):
        """
        批量提取特征
        """
        all_features = []
        
        # 获取所有图像文件
        image_files = [f for f in os.listdir(image_dir) if f.endswith('.nii.gz')]
        
        for image_file in tqdm(image_files, desc="Extracting features"):
            case_id = image_file.replace('.nii.gz', '')
            image_path = os.path.join(image_dir, image_file)
            mask_path = os.path.join(mask_dir, f"{case_id}_mask.nii.gz")
            
            if not os.path.exists(mask_path):
                print(f"Mask not found for {case_id}")
                continue
            
            features = self.extract_features_single_case(image_path, mask_path)
            
            if features:
                features['case_id'] = case_id
                all_features.append(features)
        
        # 转换为DataFrame
        df = pd.DataFrame(all_features)
        
        # 保存
        os.makedirs(os.path.dirname(output_csv), exist_ok=True)
        df.to_csv(output_csv, index=False)
        print(f"Features saved to {output_csv}")
        print(f"Total cases: {len(df)}, Total features: {len(df.columns) - 1}")
        
        return df

# 使用示例
if __name__ == "__main__":
    extractor = FeatureExtractor()
    df = extractor.extract_features_batch(
        image_dir="data/processed",
        mask_dir="data/masks",
        output_csv="results/features/radiomics_features.csv"
    )
```

#### 特征选择模块（src/feature_selection.py）

```python
import pandas as pd
import numpy as np
from sklearn.linear_model import LassoCV
from sklearn.preprocessing import StandardScaler
from sklearn.feature_selection import RFE
from sklearn.ensemble import RandomForestClassifier
import matplotlib.pyplot as plt
import seaborn as sns

class FeatureSelector:
    def __init__(self, method='lasso', n_features=15, cv_folds=5):
        self.method = method
        self.n_features = n_features
        self.cv_folds = cv_folds
        self.scaler = StandardScaler()
        self.selected_features = None
        
    def select_features_lasso(self, X, y, feature_names):
        """
        LASSO特征选择
        """
        # 标准化
        X_scaled = self.scaler.fit_transform(X)
        
        # LASSO回归
        lasso = LassoCV(cv=self.cv_folds, random_state=42, max_iter=10000)
        lasso.fit(X_scaled, y)
        
        # 获取非零系数的特征
        coef = lasso.coef_
        non_zero_idx = np.where(coef != 0)[0]
        
        # 如果特征太多，选择系数绝对值最大的n_features个
        if len(non_zero_idx) > self.n_features:
            top_idx = np.argsort(np.abs(coef))[::-1][:self.n_features]
            selected_idx = top_idx
        else:
            selected_idx = non_zero_idx
        
        self.selected_features = selected_idx
        selected_names = [feature_names[i] for i in selected_idx]
        
        print(f"Optimal alpha: {lasso.alpha_:.6f}")
        print(f"Selected {len(selected_idx)} features")
        
        # 绘制系数图
        self._plot_coefficients(coef, feature_names, selected_idx)
        
        return X_scaled[:, selected_idx], selected_names
    
    def select_features_rfe(self, X, y, feature_names):
        """
        RFE特征选择
        """
        # 标准化
        X_scaled = self.scaler.fit_transform(X)
        
        # RFE
        estimator = RandomForestClassifier(n_estimators=100, random_state=42)
        rfe = RFE(estimator, n_features_to_select=self.n_features, step=1)
        rfe.fit(X_scaled, y)
        
        selected_idx = np.where(rfe.support_)[0]
        self.selected_features = selected_idx
        selected_names = [feature_names[i] for i in selected_idx]
        
        print(f"Selected {len(selected_idx)} features")
        
        return X_scaled[:, selected_idx], selected_names
    
    def _plot_coefficients(self, coef, feature_names, selected_idx):
        """
        绘制特征系数图
        """
        selected_coef = coef[selected_idx]
        selected_names = [feature_names[i] for i in selected_idx]
        
        # 排序
        sorted_idx = np.argsort(np.abs(selected_coef))[::-1]
        
        plt.figure(figsize=(12, 8))
        plt.barh(range(len(sorted_idx)), selected_coef[sorted_idx])
        plt.yticks(range(len(sorted_idx)), [selected_names[i] for i in sorted_idx])
        plt.xlabel('Coefficient')
        plt.title('Selected Features (LASSO)')
        plt.tight_layout()
        plt.savefig('results/figures/feature_coefficients.png', dpi=300)
        plt.show()
    
    def fit_transform(self, X, y, feature_names):
        """
        拟合并转换
        """
        if self.method == 'lasso':
            return self.select_features_lasso(X, y, feature_names)
        elif self.method == 'rfe':
            return self.select_features_rfe(X, y, feature_names)
        else:
            raise ValueError(f"Unknown method: {self.method}")

# 使用示例
if __name__ == "__main__":
    # 加载特征
    df = pd.read_csv('results/features/radiomics_features.csv')
    
    # 合并临床数据
    clinical_df = pd.read_csv('data/clinical.csv')
    df = df.merge(clinical_df, on='case_id')
    
    # 准备数据
    feature_cols = [col for col in df.columns if col.startswith('original_')]
    X = df[feature_cols].values
    y = df['response'].values  # 0: 无反应, 1: 有反应
    
    # 特征选择
    selector = FeatureSelector(method='lasso', n_features=15, cv_folds=5)
    X_selected, selected_names = selector.fit_transform(X, y, feature_cols)
    
    # 保存选中的特征
    selected_df = pd.DataFrame(X_selected, columns=selected_names)
    selected_df['case_id'] = df['case_id'].values
    selected_df['response'] = y
    selected_df.to_csv('results/features/selected_features.csv', index=False)
    
    print(f"Selected features: {selected_names}")
```

#### 模型训练模块（src/model_training.py）

```python
import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split, cross_val_score, StratifiedKFold
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression
from sklearn.svm import SVC
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from xgboost import XGBClassifier
from sklearn.metrics import (
    roc_auc_score, accuracy_score, confusion_matrix,
    classification_report, roc_curve, auc
)
import matplotlib.pyplot as plt
import seaborn as sns
import yaml

class ModelTrainer:
    def __init__(self, config_path="config.yaml"):
        with open(config_path, 'r') as f:
            self.config = yaml.safe_load(f)
        
        self.models = {}
        self.results = {}
        self.scaler = StandardScaler()
        
    def prepare_data(self, df, feature_cols, label_col='response'):
        """
        准备训练数据
        """
        X = df[feature_cols].values
        y = df[label_col].values
        
        # 分割数据
        test_size = self.config['training']['test_size']
        X_train_val, X_test, y_train_val, y_test = train_test_split(
            X, y, test_size=test_size, random_state=42, stratify=y
        )
        
        val_size = self.config['training']['val_size']
        X_train, X_val, y_train, y_val = train_test_split(
            X_train_val, y_train_val, 
            test_size=val_size/(1-test_size), 
            random_state=42, 
            stratify=y_train_val
        )
        
        # 标准化
        X_train = self.scaler.fit_transform(X_train)
        X_val = self.scaler.transform(X_val)
        X_test = self.scaler.transform(X_test)
        
        print(f"Training set: {X_train.shape[0]} samples")
        print(f"Validation set: {X_val.shape[0]} samples")
        print(f"Test set: {X_test.shape[0]} samples")
        
        self.X_train = X_train
        self.X_val = X_val
        self.X_test = X_test
        self.y_train = y_train
        self.y_val = y_val
        self.y_test = y_test
        
        return X_train, X_val, X_test, y_train, y_val, y_test
    
    def train_models(self):
        """
        训练多个模型
        """
        model_configs = {
            'LR': LogisticRegression(max_iter=1000, random_state=42, class_weight='balanced'),
            'SVM': SVC(kernel='rbf', probability=True, random_state=42, class_weight='balanced'),
            'RF': RandomForestClassifier(n_estimators=500, max_depth=10, random_state=42, class_weight='balanced'),
            'XGBoost': XGBClassifier(n_estimators=500, max_depth=6, learning_rate=0.01, random_state=42)
        }
        
        for name, model in model_configs.items():
            print(f"\n{'='*50}")
            print(f"Training {name} model...")
            print(f"{'='*50}")
            
            # 交叉验证
            cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
            cv_scores = cross_val_score(model, self.X_train, self.y_train, 
                                       cv=cv, scoring='roc_auc')
            print(f"CV AUC: {cv_scores.mean():.3f} ± {cv_scores.std():.3f}")
            
            # 训练
            model.fit(self.X_train, self.y_train)
            
            # 验证集评估
            y_val_pred = model.predict(self.X_val)
            y_val_prob = model.predict_proba(self.X_val)[:, 1]
            val_auc = roc_auc_score(self.y_val, y_val_prob)
            
            # 测试集评估
            y_test_pred = model.predict(self.X_test)
            y_test_prob = model.predict_proba(self.X_test)[:, 1]
            test_auc = roc_auc_score(self.y_test, y_test_prob)
            test_acc = accuracy_score(self.y_test, y_test_pred)
            
            # 混淆矩阵
            tn, fp, fn, tp = confusion_matrix(self.y_test, y_test_pred).ravel()
            sensitivity = tp / (tp + fn)
            specificity = tn / (tn + fp)
            
            print(f"\nValidation AUC: {val_auc:.3f}")
            print(f"Test AUC: {test_auc:.3f}")
            print(f"Test Accuracy: {test_acc:.3f}")
            print(f"Test Sensitivity: {sensitivity:.3f}")
            print(f"Test Specificity: {specificity:.3f}")
            
            # 保存结果
            self.models[name] = model
            self.results[name] = {
                'cv_scores': cv_scores,
                'val_auc': val_auc,
                'test_auc': test_auc,
                'test_acc': test_acc,
                'sensitivity': sensitivity,
                'specificity': specificity,
                'y_test_pred': y_test_pred,
                'y_test_prob': y_test_prob
            }
        
        return self.models, self.results
    
    def plot_roc_curves(self, output_path='results/figures/roc_curves.png'):
        """
        绘制ROC曲线
        """
        plt.figure(figsize=(10, 8))
        
        for name, result in self.results.items():
            fpr, tpr, _ = roc_curve(self.y_test, result['y_test_prob'])
            roc_auc = auc(fpr, tpr)
            plt.plot(fpr, tpr, lw=2, label=f'{name} (AUC = {roc_auc:.3f})')
        
        plt.plot([0, 1], [0, 1], 'k--', lw=2)
        plt.xlim([0.0, 1.0])
        plt.ylim([0.0, 1.05])
        plt.xlabel('False Positive Rate', fontsize=12)
        plt.ylabel('True Positive Rate', fontsize=12)
        plt.title('ROC Curves - NSCLC Immunotherapy Response Prediction', fontsize=14)
        plt.legend(loc="lower right", fontsize=10)
        plt.grid(True, alpha=0.3)
        
        plt.savefig(output_path, dpi=300, bbox_inches='tight')
        plt.show()
    
    def get_best_model(self, metric='test_auc'):
        """
        获取最佳模型
        """
        best_score = 0
        best_model_name = None
        
        for name, result in self.results.items():
            score = result[metric]
            if score > best_score:
                best_score = score
                best_model_name = name
        
        print(f"\nBest model: {best_model_name}")
        print(f"{metric}: {best_score:.3f}")
        
        return self.models[best_model_name], best_model_name

# 使用示例
if __name__ == "__main__":
```

#### MRI预处理模块（src/mri_preprocessing.py）

```python
import SimpleITK as sitk
import numpy as np
import os

class MRIPreprocessor:
    def __init__(self):
        pass
    
    def preprocess_mri(self, image_path, output_path=None):
        """
        MRI图像预处理
        """
        # 读取图像
        image = sitk.ReadImage(image_path)
        
        # 偏置场校正
        corrector = sitk.N4BiasFieldCorrectionImageFilter()
        corrected_image = corrector.Execute(image)
        
        # 重采样到1mm各向同性
        original_spacing = image.GetSpacing()
        target_spacing = [1.0, 1.0, 1.0]
        original_size = image.GetSize()
        
        new_size = [
            int(round(original_size[i] * (original_spacing[i] / target_spacing[i])))
            for i in range(3)
        ]
        
        resampler = sitk.ResampleImageFilter()
        resampler.SetOutputSpacing(target_spacing)
        resampler.SetSize(new_size)
        resampler.SetInterpolator(sitk.sitkBSpline)
        resampler.SetOutputOrigin(image.GetOrigin())
        resampler.SetOutputDirection(image.GetDirection())
        
        resampled_image = resampler.Execute(corrected_image)
        
        # 归一化
        normalized_image = sitk.Normalize(resampled_image)
        
        # 保存
        if output_path:
            sitk.WriteImage(normalized_image, output_path)
        
        return normalized_image
    
    def batch_preprocess(self, raw_dir, output_dir):
        """
        批量预处理
        """
        os.makedirs(output_dir, exist_ok=True)
```

#### 深度学习特征提取（src/deep_learning_features.py）

```python
import torch
import torch.nn as nn
from torchvision import models, transforms
from PIL import Image
import numpy as np
import SimpleITK as sitk
from tqdm import tqdm

class DeepFeatureExtractor:
    def __init__(self, model_name='resnet50', pretrained=True):
        """
        深度学习特征提取器
        """
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        
        # 加载预训练模型
        if model_name == 'resnet50':
            self.model = models.resnet50(pretrained=pretrained)
            self.model = nn.Sequential(*list(self.model.children())[:-1])
        
        self.model.to(self.device)
        self.model.eval()
        
        # 预处理
        self.preprocess = transforms.Compose([
            transforms.Resize(224),
            transforms.CenterCrop(224),
            transforms.ToTensor(),
            transforms.Normalize(mean=[0.485, 0.456, 0.406],
                               std=[0.229, 0.224, 0.225]),
        ])
    
    def extract_features_from_3d_image(self, image_3d):
        """
        从3D图像提取深度特征
        
        参数:
            image_3d: numpy数组 (H, W, D)
            
        返回:
            features: 深度特征向量
        """
        deep_features = []
        
        # 对每个切片提取特征
        for slice_idx in range(image_3d.shape[2]):
            slice_2d = image_3d[:, :, slice_idx]
            
            # 归一化到0-255
            slice_normalized = ((slice_2d - slice_2d.min()) / 
                               (slice_2d.max() - slice_2d.min()) * 255).astype(np.uint8)
            
            # 转换为PIL图像并复制为3通道
            slice_pil = Image.fromarray(slice_normalized).convert('RGB')
            slice_tensor = self.preprocess(slice_pil).unsqueeze(0).to(self.device)
            
            # 提取特征
            with torch.no_grad():
                features = self.model(slice_tensor)
            
            deep_features.append(features.squeeze().cpu().numpy())
        
        # 平均所有切片的特征
        deep_features = np.mean(deep_features, axis=0)
        
        return deep_features
    
    def extract_features_batch(self, image_dir, output_csv):
        """
        批量提取深度特征
        """
        import pandas as pd
        
        all_features = []
        image_files = [f for f in os.listdir(image_dir) if f.endswith('.nii.gz')]
        
        for image_file in tqdm(image_files, desc="Extracting deep features"):
            case_id = image_file.replace('.nii.gz', '')
            image_path = os.path.join(image_dir, image_file)
            
            # 读取图像
            image = sitk.ReadImage(image_path)
            image_array = sitk.GetArrayFromImage(image)
            
            # 提取特征
            features = self.extract_features_from_3d_image(image_array)
            
            # 保存
            feature_dict = {f'deep_{i}': features[i] for i in range(len(features))}
            feature_dict['case_id'] = case_id
            all_features.append(feature_dict)
        
        # 保存到CSV
        df = pd.DataFrame(all_features)
        df.to_csv(output_csv, index=False)
        print(f"Deep features saved to {output_csv}")
        
        return df

# 使用示例

        image_dir='data/processed',
```

#### 融合模型（src/fusion_model.py）

```python
import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import roc_auc_score, accuracy_score
import torch
import torch.nn as nn

class FeatureFusionModel:
    def __init__(self):
        self.scaler_rad = StandardScaler()
        self.scaler_deep = StandardScaler()
        self.model = None
        
    def load_features(self, radiomics_csv, deep_csv, clinical_csv=None):
        """
        加载特征
        """
        # 加载影像组学特征
        rad_df = pd.read_csv(radiomics_csv)
        
        # 加载深度特征
        deep_df = pd.read_csv(deep_csv)
        
        # 合并
        df = rad_df.merge(deep_df, on='case_id')
        
        # 加载临床特征
        if clinical_csv:
            clinical_df = pd.read_csv(clinical_csv)
            df = df.merge(clinical_df, on='case_id')
        
        return df
    
    def prepare_data(self, df, label_col='pCR'):
        """
        准备数据
        """
        # 分离特征
        rad_cols = [col for col in df.columns if col.startswith('original_')]
        deep_cols = [col for col in df.columns if col.startswith('deep_')]
        clinical_cols = ['age', 'gender', 'tumor_size']  # 示例
        
        X_rad = df[rad_cols].values
        X_deep = df[deep_cols].values
        X_clinical = df[clinical_cols].values if all(col in df.columns for col in clinical_cols) else None
        y = df[label_col].values
        
        # 标准化
        X_rad = self.scaler_rad.fit_transform(X_rad)
        X_deep = self.scaler_deep.fit_transform(X_deep)
        
        # 特征级融合
        if X_clinical is not None:
            X_fused = np.concatenate([X_rad, X_deep, X_clinical], axis=1)
        else:
            X_fused = np.concatenate([X_rad, X_deep], axis=1)
        
        # 分割数据
        X_train, X_test, y_train, y_test = train_test_split(
            X_fused, y, test_size=0.2, random_state=42, stratify=y
        )
        
        return X_train, X_test, y_train, y_test
    
    def train_model(self, X_train, y_train, X_test, y_test):
        """
        训练融合模型
        """
        # 尝试多个模型
        models = {
            'LR': LogisticRegression(max_iter=1000, random_state=42),
            'RF': RandomForestClassifier(n_estimators=500, random_state=42)
        }
        
        best_model = None
        best_auc = 0
        
        for name, model in models.items():
            model.fit(X_train, y_train)
            y_prob = model.predict_proba(X_test)[:, 1]
            auc = roc_auc_score(y_test, y_prob)
            
            print(f"{name}: AUC = {auc:.3f}")
            
            if auc > best_auc:
                best_auc = auc
                best_model = model
        
        self.model = best_model
        print(f"\nBest model AUC: {best_auc:.3f}")
        
        return best_model

# 使用示例
if __name__ == "__main__":
    fusion_model = FeatureFusionModel()
    
    # 加载特征
    df = fusion_model.load_features(
        radiomics_csv='results/features/radiomics_features.csv',
        deep_csv='results/features/deep_features.csv',
        clinical_csv='data/clinical.csv'
    )
    
    # 准备数据
    X_train, X_test, y_train, y_test = fusion_model.prepare_data(df, label_col='pCR')
    
    # 训练模型
    model = fusion_model.train_model(X_train, y_train, X_test, y_test)
```

```python
"""
数据集下载脚本
"""
import os
import requests
from tqdm import tqdm

class DatasetDownloader:
    def __init__(self, output_dir="data/raw"):
        self.output_dir = output_dir
        os.makedirs(output_dir, exist_ok=True)
    
    def download_file(self, url, filename):
        """
        下载文件（带进度条）
        """
        filepath = os.path.join(self.output_dir, filename)
        
        response = requests.get(url, stream=True)
        total_size = int(response.headers.get('content-length', 0))
        
        with open(filepath, 'wb') as f:
            with tqdm(total=total_size, unit='B', unit_scale=True, desc=filename) as pbar:
                for chunk in response.iter_content(chunk_size=8192):
                    if chunk:
                        f.write(chunk)
                        pbar.update(len(chunk))
        
        print(f"Downloaded: {filepath}")
    
    def download_tcia_collection(self, collection_name):
        """
        下载TCIA数据集
        """
        try:
            from tciaclient import TCIAClient
            
            tc = TCIAClient(
                baseUrl="https://services.cancerimagingarchive.net/services/v4",
                resource="TCIA"
            )
            
            # 获取患者列表
            patients = tc.get_patient(collection=collection_name)
            print(f"Found {len(patients)} patients in {collection_name}")
            
            # 下载
            output_dir = os.path.join(self.output_dir, collection_name)
            os.makedirs(output_dir, exist_ok=True)
            
            for patient in tqdm(patients, desc="Downloading patients"):
                tc.get_image(patientId=patient, downloadPath=output_dir)
            
            print(f"Downloaded {collection_name} to {output_dir}")
            
        except ImportError:
            print("Please install tcia-client: pip install tcia-client")
    
    def download_all_datasets(self):
        """
```
