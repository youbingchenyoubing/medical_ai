## 1. 业务概述

### 1.1 项目背景

**临床痛点**：
- 传统血管分析依赖人工勾画，耗时耗力（单例需30-60分钟）
- 不同医师之间的分析结果差异大，一致性差（ICC仅0.6-0.7）
- 缺乏客观、定量的血管形态学评估指标
- 治疗前后对比分析困难，难以评估治疗反应

**技术价值**：
- AI自动化分析：单例处理时间缩短至3-5分钟
- 高一致性：ICC可达0.95以上
- 客观定量：提供50+项形态学特征
- 动态评估：支持治疗前后对比分析

## 4. 分割引擎设计

### 4.1 3D U-Net架构（定制化）

#### 模型设计原理

```python
"""
定制化3D U-Net用于肿瘤与血管协同分割

特点：
1. 编码器-解码器结构
2. 跳跃连接保留空间信息
3. 多尺度特征融合
4. 深度监督
"""
import torch
import torch.nn as nn
import torch.nn.functional as F

class DoubleConv3D(nn.Module):
    """双卷积块"""
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.double_conv = nn.Sequential(
            nn.Conv3d(in_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm3d(out_channels),
            nn.ReLU(inplace=True),
            nn.Conv3d(out_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm3d(out_channels),
            nn.ReLU(inplace=True)
        )
    
    def forward(self, x):
        return self.double_conv(x)

class Down3D(nn.Module):
    """下采样模块"""
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.maxpool_conv = nn.Sequential(
            nn.MaxPool3d(2),
            DoubleConv3D(in_channels, out_channels)
        )
    
    def forward(self, x):
        return self.maxpool_conv(x)

class Up3D(nn.Module):
    """上采样模块"""
    def __init__(self, in_channels, out_channels, bilinear=False):
        super().__init__()
        if bilinear:
            # 使用双线性插值
            self.up = nn.Upsample(scale_factor=2, mode='trilinear', align_corners=True)
            self.conv = DoubleConv3D(in_channels, out_channels)
        else:
            # 使用转置卷积
            self.up = nn.ConvTranspose3d(in_channels, in_channels // 2, 
                                         kernel_size=2, stride=2)
            self.conv = DoubleConv3D(in_channels, out_channels)
    
    def forward(self, x1, x2):
        x1 = self.up(x1)
        
        # 处理尺寸不匹配
        diff_d = x2.size()[2] - x1.size()[2]
        diff_h = x2.size()[3] - x1.size()[3]
        diff_w = x2.size()[4] - x1.size()[4]
        
        x1 = F.pad(x1, [diff_w // 2, diff_w - diff_w // 2,
                        diff_h // 2, diff_h - diff_h // 2,
                        diff_d // 2, diff_d - diff_d // 2])
        
        x = torch.cat([x2, x1], dim=1)
        return self.conv(x)

class UNet3D(nn.Module):
    """
    3D U-Net for Vessel Segmentation
    
    输入: (B, 1, D, H, W) - CT/MRI volume
    输出: (B, 3, D, H, W) - 背景、肿瘤、血管
    """
    def __init__(self, in_channels=1, num_classes=3):
        super().__init__()
        
        # 编码器
        self.inc = DoubleConv3D(in_channels, 64)
        self.down1 = Down3D(64, 128)
        self.down2 = Down3D(128, 256)
        self.down3 = Down3D(256, 512)
        self.down4 = Down3D(512, 1024)
        
        # 解码器
        self.up1 = Up3D(1024, 512)
        self.up2 = Up3D(512, 256)
        self.up3 = Up3D(256, 128)
        self.up4 = Up3D(128, 64)
        
        # 输出层
        self.outc = nn.Conv3d(64, num_classes, kernel_size=1)
        
    def forward(self, x):
        # 编码器路径
        x1 = self.inc(x)      # (B, 64, D, H, W)
        x2 = self.down1(x1)   # (B, 128, D/2, H/2, W/2)
        x3 = self.down2(x2)   # (B, 256, D/4, H/4, W/4)
        x4 = self.down3(x3)   # (B, 512, D/8, H/8, W/8)
        x5 = self.down4(x4)   # (B, 1024, D/16, H/16, W/16)
        
        # 解码器路径（带跳跃连接）
        x = self.up1(x5, x4)  # (B, 512, D/8, H/8, W/8)
        x = self.up2(x, x3)   # (B, 256, D/4, H/4, W/4)
        x = self.up3(x, x2)   # (B, 128, D/2, H/2, W/2)
        x = self.up4(x, x1)   # (B, 64, D, H, W)
        
        # 输出
        logits = self.outc(x)
        return logits

# 使用示例
if __name__ == "__main__":
    model = UNet3D(in_channels=1, num_classes=3)
    
    # 测试输入
    x = torch.randn(2, 1, 64, 128, 128)
    output = model(x)
    print(f"Input shape: {x.shape}")
    print(f"Output shape: {output.shape}")
    print(f"Number of parameters: {sum(p.numel() for p in model.parameters()):,}")
```

## 5. 血管骨架化算法

### 5.1 骨架化原理

**血管骨架化（Vessel Skeletonization）**是将三维血管结构简化为一维中轴线的过程。

#### 数学定义

```
骨架（Skeleton）= 中轴线（Medial Axis）

定义：血管内部所有最大内切球球心的集合

性质：
1. 拓扑等价：保持血管的连通性
2. 居中性：位于血管中心
3. 最小性：最细的表示
```

### 5.2 临床级骨架化要求

**临床应用对骨架化的要求**：
- **准确性**：中轴线必须准确反映血管走行
- **连续性**：不能有意外的中断
- **完整性**：包括所有细小分支（≥1mm）
- **效率**：单例处理时间<30秒

### 5.3 骨架化算法实现

#### 方法1：基于形态学细化

```python
import numpy as np
from skimage.morphology import skeletonize_3d
import SimpleITK as sitk

def skeletonize_vessel_morphological(vessel_mask):
    """
    基于形态学细化的骨架化
    
    参数:
        vessel_mask: 血管二值mask (D, H, W)
    
    返回:
        skeleton: 骨架点 (N, 3)
    """
    # 确保是二值图像
    vessel_binary = (vessel_mask > 0).astype(np.uint8)
    
    # 3D骨架化
    skeleton = skeletonize_3d(vessel_binary)
    
    # 提取骨架点坐标
    skeleton_points = np.argwhere(skeleton > 0)
    
    print(f"Original vessel voxels: {np.sum(vessel_binary)}")
    print(f"Skeleton points: {len(skeleton_points)}")
    
    return skeleton_points, skeleton

# 使用示例
if __name__ == "__main__":
    # 加载血管分割结果
    vessel_image = sitk.ReadImage('vessel_segmentation.nii.gz')
    vessel_mask = sitk.GetArrayFromImage(vessel_image)
    
    # 骨架化
```

### 5.4 血管拓扑结构分析

```python
import networkx as nx
import numpy as np
from scipy.spatial import KDTree

def analyze_vessel_topology(skeleton_points):
    """
    分析血管拓扑结构
    
    参数:
        skeleton_points: 骨架点坐标 (N, 3)
    
    返回:
        graph: 血管网拓扑图
        branches: 分支列表
        junctions: 分叉点列表
    """
    # 创建KD树用于快速邻域搜索
    tree = KDTree(skeleton_points)
    
    # 创建图
    graph = nx.Graph()
    
    # 添加节点
    for i, point in enumerate(skeleton_points):
        graph.add_node(i, position=point)
    
    # 添加边（连接相邻的骨架点）
    for i, point in enumerate(skeleton_points):
        # 查找距离<=√3的邻居（26邻域）
        neighbors = tree.query_ball_point(point, r=1.8)
        
        for j in neighbors:
            if i != j:
                graph.add_edge(i, j)
    
    # 识别分叉点（度>2的节点）
    junctions = [node for node, degree in graph.degree() if degree > 2]
    
    # 识别端点（度=1的节点）
    endpoints = [node for node, degree in graph.degree() if degree == 1]
    
    # 提取分支（两个分叉点之间的路径）
    branches = []
    for junction in junctions:
        neighbors = list(graph.neighbors(junction))
        for neighbor in neighbors:
            # 从分叉点开始追踪分支
            branch = trace_branch(graph, junction, neighbor, junctions)
            if len(branch) > 1:
                branches.append(branch)
    
    print(f"Total skeleton points: {len(skeleton_points)}")
    print(f"Junctions (branching points): {len(junctions)}")
    print(f"Endpoints: {len(endpoints)}")
    print(f"Branches: {len(branches)}")
    
    return graph, branches, junctions, endpoints

def trace_branch(graph, start, next_node, junctions):
    """
    追踪单个分支
    
    参数:
        graph: 血管网图
        start: 起始节点
        next_node: 下一个节点
        junctions: 分叉点列表
    
    返回:
        branch: 分支节点列表
    """
    branch = [start, next_node]
    current = next_node
    previous = start
    
    while True:
        neighbors = list(graph.neighbors(current))
        
        # 如果到达分叉点或端点，停止
        if current in junctions or len(neighbors) == 1:
            break
        
        # 找到下一个节点（不是前一个节点）
        next_nodes = [n for n in neighbors if n != previous]
        
        if len(next_nodes) == 0:
            break
        
        next_node = next_nodes[0]
        branch.append(next_node)
        previous = current
        current = next_node
    
    return branch

# 使用示例
```

## 6. 三维形态量化特征计算

### 6.1 临床特征解读

#### 曲率特征

**临床意义**：
- 正常血管：曲率较低，走行平滑
- 肿瘤周围血管：曲率增高，走行迂曲
- 血管畸形：局灶性曲率异常增高

**临床应用**：
- 肿瘤边界识别：高曲率区域可能对应肿瘤浸润
- 疗效评估：治疗后曲率降低提示血管正常化
- 预后预测：治疗前曲率高的患者预后较差

#### 分支密度特征

**临床意义**：
- 正常组织：分支密度较低且稳定
- 肿瘤区域：分支密度显著增高（血管生成）
- 抗血管生成治疗后：分支密度下降

**临床应用**：
- 肿瘤分级：高级别肿瘤分支密度更高
- 治疗方案选择：高分支密度患者适合抗血管生成治疗
- 早期疗效预测：治疗2周后分支密度变化可预测最终疗效

### 6.2 曲率计算

#### 数学定义

```
曲率（Curvature）：血管转弯的角度

数学公式：
κ = |dT/ds| = |d²r/ds²|

其中：
- r(s): 曲线参数方程
- T: 切向量
- s: 弧长参数

对于3D空间曲线：
κ = |r' × r''| / |r'|³
```

#### 实现代码

```python
import numpy as np
from scipy.interpolate import splprep, splev
from scipy.spatial.distance import cdist

def calculate_curvature_3d(points, smooth=0.1):
    """
    计算3D曲线的曲率
    
    参数:
        points: 曲线点坐标 (N, 3)
        smooth: 平滑参数
    
    返回:
        curvatures: 每个点的曲率 (N,)
    """
    # 使用B样条拟合曲线
    try:
        tck, u = splprep([points[:, 0], points[:, 1], points[:, 2]], 
                         s=smooth, k=3)
        
        # 计算一阶导数
        du = 0.01
        u_new = np.linspace(0, 1, len(points))
        derivatives_1 = np.array(splev(u_new, tck, der=1)).T  # (N, 3)
        
        # 计算二阶导数
        derivatives_2 = np.array(splev(u_new, tck, der=2)).T  # (N, 3)
        
        # 计算曲率
        # κ = |r' × r''| / |r'|³
        cross_product = np.cross(derivatives_1, derivatives_2)
        cross_norm = np.linalg.norm(cross_product, axis=1)
        
        d1_norm = np.linalg.norm(derivatives_1, axis=1)
        d1_norm_cubed = d1_norm ** 3
        
        curvatures = cross_norm / (d1_norm_cubed + 1e-8)
        
        return curvatures
        
    except Exception as e:
        print(f"Error in curve fitting: {e}")
        return np.zeros(len(points))

def calculate_curvature_discrete(points):
    """
    离散曲率计算（不需要曲线拟合）
    
    基于Menger曲率公式：
    κ = 4A / (|P1P2| × |P2P3| × |P3P1|)
    
    其中A是三角形P1P2P3的面积
    
    参数:
        points: 曲线点坐标 (N, 3)
    
    返回:
        curvatures: 曲率 (N,)
    """
    n = len(points)
    curvatures = np.zeros(n)
    
    for i in range(1, n - 1):
        p1 = points[i - 1]
        p2 = points[i]
        p3 = points[i + 1]
        
        # 计算边长
        a = np.linalg.norm(p2 - p1)
        b = np.linalg.norm(p3 - p2)
        c = np.linalg.norm(p3 - p1)
        
        # 海伦公式计算三角形面积
        s = (a + b + c) / 2
        area = np.sqrt(max(s * (s - a) * (s - b) * (s - c), 0))
        
        # Menger曲率
        if a * b * c > 0:
            curvatures[i] = 4 * area / (a * b * c)
        else:
            curvatures[i] = 0
    
    # 边界点
    curvatures[0] = curvatures[1]
    curvatures[-1] = curvatures[-2]
    
    return curvatures

# 使用示例
```

### 6.3 扭率计算

#### 数学定义

```
扭率（Torsion）：血管在三维空间中"拧"的程度

数学公式：
τ = (r' × r'') · r''' / |r' × r''|²

物理意义：
- τ = 0: 平面曲线（不扭曲）
- τ > 0: 右手螺旋
- τ < 0: 左手螺旋
```

#### 实现代码

```python
import numpy as np
from scipy.interpolate import splprep, splev

def calculate_torsion_3d(points, smooth=0.1):
    """
    计算3D曲线的扭率
    
    参数:
        points: 曲线点坐标 (N, 3)
        smooth: 平滑参数
    
    返回:
        torsions: 每个点的扭率 (N,)
    """
    try:
        # B样条拟合
        tck, u = splprep([points[:, 0], points[:, 1], points[:, 2]], 
                         s=smooth, k=3)
        
        u_new = np.linspace(0, 1, len(points))
        
        # 一阶、二阶、三阶导数
        d1 = np.array(splev(u_new, tck, der=1)).T  # r'
        d2 = np.array(splev(u_new, tck, der=2)).T  # r''
        d3 = np.array(splev(u_new, tck, der=3)).T  # r'''
        
        # 计算扭率
        # τ = (r' × r'') · r''' / |r' × r''|²
        cross_d1_d2 = np.cross(d1, d2)
        cross_norm_sq = np.sum(cross_d1_d2 ** 2, axis=1)
        
        torsions = np.sum(cross_d1_d2 * d3, axis=1) / (cross_norm_sq + 1e-8)
        
        return torsions
        
    except Exception as e:
        print(f"Error in torsion calculation: {e}")
        return np.zeros(len(points))

def calculate_torsion_discrete(points):
    """
    离散扭率计算
    
    参数:
        points: 曲线点坐标 (N, 3)
    
    返回:
        torsions: 扭率 (N,)
    """
    n = len(points)
    torsions = np.zeros(n)
    
    for i in range(2, n - 2):
        # 使用5点模板
        p0, p1, p2, p3, p4 = points[i-2:i+3]
        
        # 中心差分近似导数
        d1 = (-p4 + 4*p3 - 6*p2 + 4*p1 - p0) / 2  # r'
        d2 = (p4 - 2*p3 + p2)  # r''
        d3 = (p4 - 2*p3 + p2)  # r''' (简化)
        
        # 计算扭率
        cross = np.cross(d1, d2)
        cross_norm_sq = np.sum(cross ** 2)
        
        if cross_norm_sq > 1e-8:
            torsions[i] = np.dot(cross, d3) / cross_norm_sq
    
    return torsions
```

### 6.4 分支密度计算

#### 定义

```
分支密度（Branching Density）：单位体积内血管分叉的数量

计算方法：
Branching Density = Number of Junctions / Tumor Volume

单位：分叉数/mm³
```

#### 实现代码

```python
import numpy as np
from scipy.spatial import ConvexHull

def calculate_branching_density(junctions, tumor_mask, spacing=(1.0, 1.0, 1.0)):
    """
    计算分支密度
    
    参数:
        junctions: 分叉点坐标列表 [(x, y, z), ...]
        tumor_mask: 肿瘤二值mask (D, H, W)
        spacing: 体素间距 (mm)
    
    返回:
        density: 分支密度 (分叉数/mm³)
        tumor_volume: 肿瘤体积 (mm³)
    """
    # 计算肿瘤体积
    voxel_volume = spacing[0] * spacing[1] * spacing[2]
    tumor_volume = np.sum(tumor_mask > 0) * voxel_volume
    
    # 分叉点数量
    num_junctions = len(junctions)
    
    # 分支密度
    density = num_junctions / tumor_volume if tumor_volume > 0 else 0
    
    print(f"Number of junctions: {num_junctions}")
    print(f"Tumor volume: {tumor_volume:.2f} mm³")
    print(f"Branching density: {density:.4f} junctions/mm³")
    
    return density, tumor_volume

def calculate_local_branching_density(junctions, tumor_mask, spacing=(1.0, 1.0, 1.0), 
                                       radius=10.0):
    """
    计算局部分支密度（空间变化）
    
    参数:
        junctions: 分叉点坐标列表
        tumor_mask: 肿瘤mask
        spacing: 体素间距
        radius: 局部邻域半径 (mm)
    
    返回:
        density_map: 局部分支密度图
    """
    from scipy.ndimage import gaussian_filter
    
    # 创建分叉点密度图
    junction_map = np.zeros_like(tumor_mask, dtype=np.float32)
    
    for junction in junctions:
        # 转换为体素坐标
        z, y, x = junction.astype(int)
        if 0 <= z < junction_map.shape[0] and \
           0 <= y < junction_map.shape[1] and \
           0 <= x < junction_map.shape[2]:
            junction_map[z, y, x] = 1
    
    # 高斯平滑（模拟局部邻域）
    sigma = radius / np.array(spacing)
    density_map = gaussian_filter(junction_map, sigma=sigma)
    
    # 归一化到体积
    voxel_volume = spacing[0] * spacing[1] * spacing[2]
    sphere_volume = (4/3) * np.pi * radius**3
    density_map = density_map / sphere_volume
    
    return density_map

def calculate_vessel_density(vessel_mask, tumor_mask, spacing=(1.0, 1.0, 1.0)):
    """
    计算血管密度
    
    参数:
        vessel_mask: 血管mask
        tumor_mask: 肿瘤mask
        spacing: 体素间距
    
    返回:
        vessel_density: 血管密度 (血管体积/肿瘤体积)
    """
    voxel_volume = spacing[0] * spacing[1] * spacing[2]
    
    vessel_volume = np.sum(vessel_mask > 0) * voxel_volume
    tumor_volume = np.sum(tumor_mask > 0) * voxel_volume
    
    vessel_density = vessel_volume / tumor_volume if tumor_volume > 0 else 0
    
    print(f"Vessel volume: {vessel_volume:.2f} mm³")
    print(f"Tumor volume: {tumor_volume:.2f} mm³")
    print(f"Vessel density: {vessel_density:.4f}")
    
    return vessel_density
```

### 6.5 综合特征提取

```python
import numpy as np
import pandas as pd
import SimpleITK as sitk
from typing import Dict, List, Tuple

class VesselMorphometryExtractor:
    """
    血管形态学特征提取器
    
    提取的特征：
    1. 曲率特征：mean, max, std, percentiles
    2. 扭率特征：mean, max, std, percentiles
    3. 分支特征：density, total_count, mean_distance
    4. 血管密度：global, local
    5. 半径特征：mean, max, min, std
    """
    
    def __init__(self, spacing=(1.0, 1.0, 1.0)):
        self.spacing = spacing
    
    def extract_all_features(self, 
                            vessel_mask: np.ndarray,
                            tumor_mask: np.ndarray,
                            skeleton_points: np.ndarray,
                            branches: List,
                            junctions: List) -> Dict:
        """
        提取所有血管形态学特征
        
        参数:
            vessel_mask: 血管分割mask
            tumor_mask: 肿瘤分割mask
            skeleton_points: 骨架点
            branches: 分支列表
            junctions: 分叉点列表
        
        返回:
            features: 特征字典
        """
        features = {}
        
        # 1. 曲率特征
        print("Extracting curvature features...")
        curvature_features = self._extract_curvature_features(branches)
        features.update(curvature_features)
        
        # 2. 扭率特征
        print("Extracting torsion features...")
        torsion_features = self._extract_torsion_features(branches)
        features.update(torsion_features)
        
        # 3. 分支特征
        print("Extracting branching features...")
        branching_features = self._extract_branching_features(
            junctions, tumor_mask, branches
        )
        features.update(branching_features)
        
        # 4. 血管密度特征
        print("Extracting vessel density features...")
        density_features = self._extract_density_features(
            vessel_mask, tumor_mask
        )
        features.update(density_features)
        
        # 5. 半径特征
        print("Extracting radius features...")
        radius_features = self._extract_radius_features(
            vessel_mask, skeleton_points
        )
        features.update(radius_features)
        
        # 6. 拓扑特征
        print("Extracting topology features...")
        topology_features = self._extract_topology_features(
            branches, junctions, skeleton_points
        )
        features.update(topology_features)
        
        return features
    
    def _extract_curvature_features(self, branches: List) -> Dict:
        """提取曲率特征"""
        all_curvatures = []
        
        for branch in branches:

    def _extract_torsion_features(self, branches: List) -> Dict:
        """提取扭率特征"""

        for branch in branches:
            if len(branch) > 4:
                branch_points = skeleton_points[branch]
                torsions = calculate_torsion_3d(branch_points)
                all_torsions.extend(torsions)
        
        all_torsions = np.array(all_torsions)
        
        features = {
            'torsion_mean': np.mean(np.abs(all_torsions)),
            'torsion_std': np.std(all_torsions),
            'torsion_abs_mean': np.mean(np.abs(all_torsions)),
            'torsion_max': np.max(np.abs(all_torsions)),

    def _extract_density_features(self,
```
