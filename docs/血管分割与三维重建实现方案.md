## 2. 分割引擎设计

### 2.1 3D U-Net架构（定制化）

#### 模型设计原理

```python
"""
定制化3D U-Net用于肿瘤与血管协同分割

特点：
1. 编码器-解码器结构
2. 跳跃连接保留空间信息
3. 多尺度特征融合
4. 深度监督
"""
import torch
import torch.nn as nn
import torch.nn.functional as F

class DoubleConv3D(nn.Module):
    """双卷积块"""
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.double_conv = nn.Sequential(
            nn.Conv3d(in_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm3d(out_channels),
            nn.ReLU(inplace=True),
            nn.Conv3d(out_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm3d(out_channels),
            nn.ReLU(inplace=True)
        )
    
    def forward(self, x):
        return self.double_conv(x)

class Down3D(nn.Module):
    """下采样模块"""
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.maxpool_conv = nn.Sequential(
            nn.MaxPool3d(2),
            DoubleConv3D(in_channels, out_channels)
        )
    
    def forward(self, x):
        return self.maxpool_conv(x)

class Up3D(nn.Module):
    """上采样模块"""
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.up = nn.ConvTranspose3d(in_channels, in_channels // 2, 
                                      kernel_size=2, stride=2)
        self.conv = DoubleConv3D(in_channels, out_channels)
    
    def forward(self, x1, x2):
        x1 = self.up(x1)
        
        # 处理尺寸不匹配
        diff_d = x2.size()[2] - x1.size()[2]
        diff_h = x2.size()[3] - x1.size()[3]
        diff_w = x2.size()[4] - x1.size()[4]
        
        x1 = F.pad(x1, [diff_w // 2, diff_w - diff_w // 2,
                        diff_h // 2, diff_h - diff_h // 2,
                        diff_d // 2, diff_d - diff_d // 2])
        
        x = torch.cat([x2, x1], dim=1)
        return self.conv(x)

class UNet3D(nn.Module):
    """
    3D U-Net for Vessel Segmentation
    
    输入: (B, 1, D, H, W) - CT/MRI volume
    输出: (B, 3, D, H, W) - 背景、肿瘤、血管
    """
    def __init__(self, in_channels=1, num_classes=3):
        super().__init__()
        
        # 编码器
        self.inc = DoubleConv3D(in_channels, 64)
        self.down1 = Down3D(64, 128)
        self.down2 = Down3D(128, 256)
        self.down3 = Down3D(256, 512)
        self.down4 = Down3D(512, 1024)
        
        # 解码器
        self.up1 = Up3D(1024, 512)
        self.up2 = Up3D(512, 256)
        self.up3 = Up3D(256, 128)
        self.up4 = Up3D(128, 64)
        
        # 输出层
        self.outc = nn.Conv3d(64, num_classes, kernel_size=1)
        
    def forward(self, x):
        # 编码器路径
        x1 = self.inc(x)
        x2 = self.down1(x1)
        x3 = self.down2(x2)
        x4 = self.down3(x3)
        x5 = self.down4(x4)
        
        # 解码器路径（带跳跃连接）
        x = self.up1(x5, x4)
        x = self.up2(x, x3)
        x = self.up3(x, x2)
        x = self.up4(x, x1)
        
        # 输出
        logits = self.outc(x)
        return logits

# 使用示例
if __name__ == "__main__":
    model = UNet3D(in_channels=1, num_classes=3)
    
    # 测试输入
    x = torch.randn(2, 1, 64, 128, 128)
    output = model(x)
    print(f"Input shape: {x.shape}")
    print(f"Output shape: {output.shape}")
    print(f"Number of parameters: {sum(p.numel() for p in model.parameters()):,}")
```

### 2.2 nnU-Net（自适应配置）

#### nnU-Net核心思想

```python
"""
nnU-Net (no-new-Net)
自适应配置的U-Net，自动优化：
1. 网络架构（2D/3D U-Net）
2. Batch size
3. Patch size
4. 数据预处理策略
"""

# 使用MONAI实现nnU-Net
from monai.networks.nets import UNet
from monai.losses import DiceLoss
from monai.metrics import DiceMetric

class nnUNetSegmenter:
    def __init__(self, 
                 spatial_dims=3,
                 in_channels=1,
                 out_channels=3,
                 img_size=(64, 128, 128)):
        """
        nnU-Net风格分割器
        
        参数:
            spatial_dims: 空间维度（2或3）
            in_channels: 输入通道数
            out_channels: 输出类别数（背景、肿瘤、血管）
            img_size: 输入图像大小
        """
        self.spatial_dims = spatial_dims
        self.img_size = img_size
        
        # 自动计算网络参数
        self.channels = self._auto_configure_channels()
        self.strides = self._auto_configure_strides()
        
        # 创建模型
        self.model = UNet(
            spatial_dims=spatial_dims,
            in_channels=in_channels,
            out_channels=out_channels,
            channels=self.channels,
            strides=self.strides,
            num_res_units=2,  # 残差连接
            norm='batch',
            dropout=0.1
        )
        
        # 损失函数（多类别Dice）
        self.loss_function = DiceLoss(
            to_onehot_y=True,
            softmax=True,
            include_background=False
        )
        
        # 评估指标
        self.dice_metric = DiceMetric(
            include_background=False,
            reduction="mean"
        )
    
    def _auto_configure_channels(self):
        """
        根据图像大小自动配置通道数
        nnU-Net的核心：自适应配置
        """
        # 基于图像大小计算网络深度
        min_dim = min(self.img_size)
        
        if min_dim >= 128:
            # 大图像：更深网络
            channels = (32, 64, 128, 256, 512, 512)
        elif min_dim >= 64:
            # 中等图像
            channels = (32, 64, 128, 256, 512)
        else:
            # 小图像
            channels = (32, 64, 128, 256)
        
        return channels
    
    def _auto_configure_strides(self):
        """自动配置步长"""
        n_layers = len(self.channels) - 1
        return (2,) * n_layers
    
    def train_step(self, batch_data):
        """
        训练步骤
        
        参数:
            batch_data: {'image': tensor, 'label': tensor}
        """
        self.model.train()
        
        images = batch_data['image']
        labels = batch_data['label']
        
        # 前向传播
        outputs = self.model(images)
        
        # 计算损失
        loss = self.loss_function(outputs, labels)
        
        return loss
    
    def inference(self, image):
        """
        推理
        
        参数:
            image: 输入图像 tensor (B, C, D, H, W)
        """
        self.model.eval()
        
        with torch.no_grad():
            # 滑动窗口推理（处理大图像）
            from monai.inferers import sliding_window_inference
            
            outputs = sliding_window_inference(
                inputs=image,
                roi_size=self.img_size,
                sw_batch_size=4,
                predictor=self.model,
                overlap=0.5
            )
            
            # 获取预测类别
            preds = torch.argmax(outputs, dim=1, keepdim=True)
            
        return preds, outputs

# 使用示例
if __name__ == "__main__":
    # 初始化
    segmenter = nnUNetSegmenter(
        spatial_dims=3,
        in_channels=1,
        out_channels=3,  # 背景、肿瘤、血管
        img_size=(64, 128, 128)
    )
    
    print(f"Model channels: {segmenter.channels}")
    print(f"Model strides: {segmenter.strides}")
```

### 2.3 肿瘤与血管协同分割

#### 多任务学习策略

```python
"""
肿瘤与血管协同分割（Co-segmentation）

策略：
1. 多任务学习：同时分割肿瘤和血管
2. 共享编码器，独立解码器
3. 损失函数加权
"""
import torch
import torch.nn as nn
from monai.losses import DiceLoss, DiceCELoss

class CoSegmentationNet(nn.Module):
    """
    协同分割网络
    同时分割肿瘤和血管
    """
    def __init__(self, in_channels=1):
        super().__init__()
        
        # 共享编码器
        self.encoder = nn.Sequential(
            nn.Conv3d(in_channels, 64, 3, padding=1),
            nn.BatchNorm3d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool3d(2),
            
            nn.Conv3d(64, 128, 3, padding=1),
            nn.BatchNorm3d(128),
            nn.ReLU(inplace=True),
            nn.MaxPool3d(2),
            
            nn.Conv3d(128, 256, 3, padding=1),
            nn.BatchNorm3d(256),
            nn.ReLU(inplace=True),
            nn.MaxPool3d(2),
            
            nn.Conv3d(256, 512, 3, padding=1),
            nn.BatchNorm3d(512),
            nn.ReLU(inplace=True),
        )
        
        # 肿瘤分割解码器
        self.tumor_decoder = nn.Sequential(
            nn.ConvTranspose3d(512, 256, 2, stride=2),
            nn.Conv3d(256, 256, 3, padding=1),
            nn.BatchNorm3d(256),
            nn.ReLU(inplace=True),
            
            nn.ConvTranspose3d(256, 128, 2, stride=2),
            nn.Conv3d(128, 128, 3, padding=1),
            nn.BatchNorm3d(128),
            nn.ReLU(inplace=True),
            
            nn.ConvTranspose3d(128, 64, 2, stride=2),
            nn.Conv3d(64, 64, 3, padding=1),
            nn.BatchNorm3d(64),
            nn.ReLU(inplace=True),
            
            nn.Conv3d(64, 2, 1)  # 背景+肿瘤
        )
        
        # 血管分割解码器
        self.vessel_decoder = nn.Sequential(
            nn.ConvTranspose3d(512, 256, 2, stride=2),
            nn.Conv3d(256, 256, 3, padding=1),
            nn.BatchNorm3d(256),
            nn.ReLU(inplace=True),
            
            nn.ConvTranspose3d(256, 128, 2, stride=2),
            nn.Conv3d(128, 128, 3, padding=1),
            nn.BatchNorm3d(128),
            nn.ReLU(inplace=True),
            
            nn.ConvTranspose3d(128, 64, 2, stride=2),
            nn.Conv3d(64, 64, 3, padding=1),
            nn.BatchNorm3d(64),
            nn.ReLU(inplace=True),
            
            nn.Conv3d(64, 2, 1)  # 背景+血管
        )
    
    def forward(self, x):
        # 共享特征提取
        features = self.encoder(x)
        
        # 肿瘤分割
        tumor_logits = self.tumor_decoder(features)
        
        # 血管分割
        vessel_logits = self.vessel_decoder(features)
        
        return tumor_logits, vessel_logits

class CoSegmentationLoss(nn.Module):
    """
    协同分割损失函数
    """
    def __init__(self, tumor_weight=1.0, vessel_weight=1.0):
        super().__init__()
        self.tumor_weight = tumor_weight
        self.vessel_weight = vessel_weight
        
        self.dice_loss = DiceLoss(
            to_onehot_y=True,
            softmax=True,
            include_background=False
        )
    
    def forward(self, tumor_pred, tumor_label, vessel_pred, vessel_label):
        # 肿瘤损失
        tumor_loss = self.dice_loss(tumor_pred, tumor_label)
        
        # 血管损失
        vessel_loss = self.dice_loss(vessel_pred, vessel_label)
        
        # 总损失
        total_loss = (self.tumor_weight * tumor_loss + 
                      self.vessel_weight * vessel_loss)
        
        return total_loss, tumor_loss, vessel_loss
```

## 3. 血管骨架化算法

### 3.1 骨架化原理

**血管骨架化（Vessel Skeletonization）**是将三维血管结构简化为一维中轴线的过程。

#### 数学定义

```
骨架（Skeleton）= 中轴线（Medial Axis）

定义：血管内部所有最大内切球球心的集合

性质：
1. 拓扑等价：保持血管的连通性
2. 居中性：位于血管中心
3. 最小性：最细的表示
```

### 3.2 骨架化算法实现

#### 方法1：基于形态学细化

```python
import numpy as np
from skimage.morphology import skeletonize_3d
import SimpleITK as sitk

def skeletonize_vessel_morphological(vessel_mask):
    """
    基于形态学细化的骨架化
    
    参数:

    print(f"Original vessel voxels: {np.sum(vessel_binary)}")
    print(f"Skeleton points: {len(skeleton_points)}")
    
    return skeleton_points, skeleton
```

#### 方法2：基于距离变换

```python
from scipy.ndimage import distance_transform_edt
from skimage.morphology import skeletonize_3d
import numpy as np

def skeletonize_vessel_distance_transform(vessel_mask):
    """
    基于距离变换的骨架化
    
    优点：
    1. 更准确的中轴线
    2. 保留血管半径信息
    
    参数:
        vessel_mask: 血管二值mask
    
    返回:
        skeleton: 骨架
        distance_map: 距离图（包含半径信息）
    """
    # 计算距离变换
    distance_map = distance_transform_edt(vessel_mask)
    
    # 骨架化
    skeleton = skeletonize_3d(vessel_mask)
    
    # 在骨架点上提取半径
    skeleton_points = np.argwhere(skeleton > 0)
    radii = distance_map[skeleton[:, :, :] > 0]
    
    # 创建带半径的骨架
    skeleton_with_radius = np.zeros_like(vessel_mask, dtype=np.float32)
    for i, point in enumerate(skeleton_points):
        skeleton_with_radius[tuple(point)] = radii[i]
    
    return skeleton, distance_map, skeleton_with_radius

# 使用示例
skeleton, distance_map, skeleton_radius = skeletonize_vessel_distance_transform(vessel_mask)
```

```python
import vtk
import numpy as np
from vtk.util.numpy_support import vtk_to_numpy, numpy_to_vtk

def skeletonize_vessel_vtk(vessel_mask, spacing=(1.0, 1.0, 1.0)):
    """
    使用VTK进行高级骨架化
    
    优点：
    1. 高质量的中轴线提取
    2. 自动处理分支
    3. 输出拓扑结构
    
    参数:
        vessel_mask: 血管二值mask (D, H, W)
        spacing: 体素间距 (mm)
    
    返回:
        centerline_points: 中轴线点 (N, 3)
        centerline_graph: 拓扑图结构
    """
    # 转换为VTK图像
    vtk_image = vtk.vtkImageData()
    vtk_image.SetDimensions(vessel_mask.shape[::-1])
    vtk_image.SetSpacing(spacing)
    
    # 转换数据
    flat_mask = vessel_mask.flatten(order='F').astype(np.uint8)
    vtk_array = numpy_to_vtk(flat_mask)
    vtk_array.SetName('mask')
    vtk_image.GetPointData().SetScalars(vtk_array)
    
    # 骨架化过滤器
    skeleton_filter = vtk.vtkImageSkeleton2D()
    skeleton_filter.SetInputData(vtk_image)
    skeleton_filter.SetThinning(vtk.vtkImageSkeleton2D.VTK_SKELETON_THINNING)
    skeleton_filter.Update()
    
    # 提取结果
    skeleton_output = skeleton_filter.GetOutput()
    skeleton_array = vtk_to_numpy(skeleton_output.GetPointData().GetScalars())
    skeleton_array = skeleton_array.reshape(vessel_mask.shape[::-1], order='F')
    
    # 提取骨架点
    skeleton_points = np.argwhere(skeleton_array > 0)
    
    return skeleton_points, skeleton_array
```

### 3.3 血管拓扑结构分析

```python
import networkx as nx
import numpy as np
from scipy.spatial import KDTree

def analyze_vessel_topology(skeleton_points):
    """
    分析血管拓扑结构
    
    参数:
        skeleton_points: 骨架点坐标 (N, 3)
    
    返回:
        graph: 血管网拓扑图
        branches: 分支列表
        junctions: 分叉点列表
    """
    # 创建KD树用于快速邻域搜索
    tree = KDTree(skeleton_points)
    
    # 创建图
    graph = nx.Graph()
    
    # 添加节点
    for i, point in enumerate(skeleton_points):
        graph.add_node(i, position=point)
    
    # 添加边（连接相邻的骨架点）
    for i, point in enumerate(skeleton_points):
        # 查找距离<=√3的邻居（26邻域）
        neighbors = tree.query_ball_point(point, r=1.8)
        
        for j in neighbors:
            if i != j:
                graph.add_edge(i, j)
    
    # 识别分叉点（度>2的节点）
    junctions = [node for node, degree in graph.degree() if degree > 2]
    
    # 识别端点（度=1的节点）
    endpoints = [node for node, degree in graph.degree() if degree == 1]
    
    # 提取分支（两个分叉点之间的路径）
    branches = []
    for junction in junctions:
        neighbors = list(graph.neighbors(junction))
        for neighbor in neighbors:
            # 从分叉点开始追踪分支
            branch = trace_branch(graph, junction, neighbor, junctions)
            if len(branch) > 1:
                branches.append(branch)
    
    print(f"Total skeleton points: {len(skeleton_points)}")
    print(f"Junctions (branching points): {len(junctions)}")
    print(f"Endpoints: {len(endpoints)}")
    print(f"Branches: {len(branches)}")
    
    return graph, branches, junctions, endpoints

def trace_branch(graph, start, next_node, junctions):
    """
    追踪单个分支
    
    参数:
        graph: 血管网图
        start: 起始节点
        next_node: 下一个节点
        junctions: 分叉点列表
    
    返回:
        branch: 分支节点列表
    """
    branch = [start, next_node]
    current = next_node
    previous = start
    
    while True:
        neighbors = list(graph.neighbors(current))
        
        # 如果到达分叉点或端点，停止
        if current in junctions or len(neighbors) == 1:
            break
        
        # 找到下一个节点（不是前一个节点）
        next_nodes = [n for n in neighbors if n != previous]
        
        if len(next_nodes) == 0:
            break
        
        next_node = next_nodes[0]
        branch.append(next_node)
        previous = current
        current = next_node
    
    return branch

# 使用示例
```

## 4. 三维形态量化特征计算

### 4.1 曲率计算

#### 数学定义

```
曲率（Curvature）：血管转弯的角度

数学公式：
κ = |dT/ds| = |d²r/ds²|

其中：
- r(s): 曲线参数方程
- T: 切向量
- s: 弧长参数

对于3D空间曲线：
κ = |r' × r''| / |r'|³
```

#### 实现代码

```python
import numpy as np
from scipy.interpolate import splprep, splev
from scipy.spatial.distance import cdist

def calculate_curvature_3d(points, smooth=0.1):
    """
    计算3D曲线的曲率
    
    参数:
        points: 曲线点坐标 (N, 3)
        smooth: 平滑参数
    
    返回:
        curvatures: 每个点的曲率 (N,)
    """
    # 使用B样条拟合曲线
    try:
        tck, u = splprep([points[:, 0], points[:, 1], points[:, 2]], 
                         s=smooth, k=3)
        
        # 计算一阶导数
        du = 0.01
        u_new = np.linspace(0, 1, len(points))
        derivatives_1 = np.array(splev(u_new, tck, der=1)).T  # (N, 3)
        
        # 计算二阶导数
        derivatives_2 = np.array(splev(u_new, tck, der=2)).T  # (N, 3)
        
        # 计算曲率
        # κ = |r' × r''| / |r'|³
        cross_product = np.cross(derivatives_1, derivatives_2)
        cross_norm = np.linalg.norm(cross_product, axis=1)
        
        d1_norm = np.linalg.norm(derivatives_1, axis=1)
        d1_norm_cubed = d1_norm ** 3
        
        curvatures = cross_norm / (d1_norm_cubed + 1e-8)
        
        return curvatures
        
    except Exception as e:
        print(f"Error in curve fitting: {e}")
        return np.zeros(len(points))

def calculate_curvature_discrete(points):
    """
    离散曲率计算（不需要曲线拟合）
    
    基于Menger曲率公式：
    κ = 4A / (|P1P2| × |P2P3| × |P3P1|)
    
    其中A是三角形P1P2P3的面积
    
    参数:
        points: 曲线点坐标 (N, 3)
    
    返回:
        curvatures: 曲率 (N,)
    """
    n = len(points)
    curvatures = np.zeros(n)
    
    for i in range(1, n - 1):
        p1 = points[i - 1]
        p2 = points[i]
        p3 = points[i + 1]
        
        # 计算边长
        a = np.linalg.norm(p2 - p1)
        b = np.linalg.norm(p3 - p2)
        c = np.linalg.norm(p3 - p1)
        
        # 海伦公式计算三角形面积
        s = (a + b + c) / 2
        area = np.sqrt(max(s * (s - a) * (s - b) * (s - c), 0))
        
        # Menger曲率
        if a * b * c > 0:
            curvatures[i] = 4 * area / (a * b * c)
        else:
            curvatures[i] = 0
    
    # 边界点
    curvatures[0] = curvatures[1]
    curvatures[-1] = curvatures[-2]
    
    return curvatures
```

### 4.2 扭率计算

#### 数学定义

```
扭率（Torsion）：血管在三维空间中"拧"的程度

数学公式：
τ = (r' × r'') · r''' / |r' × r''|²

物理意义：
- τ = 0: 平面曲线（不扭曲）
- τ > 0: 右手螺旋
- τ < 0: 左手螺旋
```

#### 实现代码

```python
import numpy as np
from scipy.interpolate import splprep, splev

def calculate_torsion_3d(points, smooth=0.1):
    """
    计算3D曲线的扭率
    
    参数:
        points: 曲线点坐标 (N, 3)
        smooth: 平滑参数
    
    返回:
        torsions: 每个点的扭率 (N,)
    """
    try:
        # B样条拟合
        tck, u = splprep([points[:, 0], points[:, 1], points[:, 2]], 
                         s=smooth, k=3)
        
        u_new = np.linspace(0, 1, len(points))
        
        # 一阶、二阶、三阶导数
        d1 = np.array(splev(u_new, tck, der=1)).T  # r'
        d2 = np.array(splev(u_new, tck, der=2)).T  # r''
        d3 = np.array(splev(u_new, tck, der=3)).T  # r'''
        
        # 计算扭率
        # τ = (r' × r'') · r''' / |r' × r''|²
        cross_d1_d2 = np.cross(d1, d2)
        cross_norm_sq = np.sum(cross_d1_d2 ** 2, axis=1)
        
        torsions = np.sum(cross_d1_d2 * d3, axis=1) / (cross_norm_sq + 1e-8)
        
        return torsions
        
    except Exception as e:
        print(f"Error in torsion calculation: {e}")
        return np.zeros(len(points))

def calculate_torsion_discrete(points):
    """
    离散扭率计算
    
    参数:
        points: 曲线点坐标 (N, 3)
    
    返回:
        torsions: 扭率 (N,)
    """
    n = len(points)
    torsions = np.zeros(n)
    
    for i in range(2, n - 2):
        # 使用5点模板
        p0, p1, p2, p3, p4 = points[i-2:i+3]
        
        # 中心差分近似导数
        d1 = (-p4 + 4*p3 - 6*p2 + 4*p1 - p0) / 2  # r'
        d2 = (p4 - 2*p3 + p2)  # r''
        d3 = (p4 - 2*p3 + p2)  # r''' (简化)
        
        # 计算扭率
        cross = np.cross(d1, d2)
        cross_norm_sq = np.sum(cross ** 2)
        
        if cross_norm_sq > 1e-8:
            torsions[i] = np.dot(cross, d3) / cross_norm_sq
    
    return torsions
```

### 4.3 分支密度计算

#### 定义

```
分支密度（Branching Density）：单位体积内血管分叉的数量

计算方法：
Branching Density = Number of Junctions / Tumor Volume

单位：分叉数/mm³
```

#### 实现代码

```python
import numpy as np
from scipy.spatial import ConvexHull

def calculate_branching_density(junctions, tumor_mask, spacing=(1.0, 1.0, 1.0)):
    """
    计算分支密度
    
    参数:
        junctions: 分叉点坐标列表 [(x, y, z), ...]
        tumor_mask: 肿瘤二值mask (D, H, W)
        spacing: 体素间距 (mm)
    
    返回:
        density: 分支密度 (分叉数/mm³)
        tumor_volume: 肿瘤体积 (mm³)
    """
    # 计算肿瘤体积
    voxel_volume = spacing[0] * spacing[1] * spacing[2]
    tumor_volume = np.sum(tumor_mask > 0) * voxel_volume
    
    # 分叉点数量
    num_junctions = len(junctions)
    
    # 分支密度
    density = num_junctions / tumor_volume if tumor_volume > 0 else 0
    
    print(f"Number of junctions: {num_junctions}")
    print(f"Tumor volume: {tumor_volume:.2f} mm³")
    print(f"Branching density: {density:.4f} junctions/mm³")
    
    return density, tumor_volume

def calculate_local_branching_density(junctions, tumor_mask, spacing=(1.0, 1.0, 1.0), 
                                       radius=10.0):
    """
    计算局部分支密度（空间变化）
    
    参数:
        junctions: 分叉点坐标列表
        tumor_mask: 肿瘤mask
        spacing: 体素间距
        radius: 局部邻域半径 (mm)
    
    返回:
        density_map: 局部分支密度图
    """
    from scipy.ndimage import gaussian_filter
    
    # 创建分叉点密度图
    junction_map = np.zeros_like(tumor_mask, dtype=np.float32)
    
    for junction in junctions:
        # 转换为体素坐标
        z, y, x = junction.astype(int)
        if 0 <= z < junction_map.shape[0] and \
           0 <= y < junction_map.shape[1] and \
           0 <= x < junction_map.shape[2]:
            junction_map[z, y, x] = 1
    
    # 高斯平滑（模拟局部邻域）
    sigma = radius / np.array(spacing)
    density_map = gaussian_filter(junction_map, sigma=sigma)
    
    # 归一化到体积
    voxel_volume = spacing[0] * spacing[1] * spacing[2]
    sphere_volume = (4/3) * np.pi * radius**3
    density_map = density_map / sphere_volume
    
    return density_map

def calculate_vessel_density(vessel_mask, tumor_mask, spacing=(1.0, 1.0, 1.0)):
    """
    计算血管密度
    
    参数:
        vessel_mask: 血管mask
        tumor_mask: 肿瘤mask
        spacing: 体素间距
    
    返回:
        vessel_density: 血管密度 (血管体积/肿瘤体积)

    tumor_volume = np.sum(tumor_mask > 0) * voxel_volume
    
    vessel_density = vessel_volume / tumor_volume if tumor_volume > 0 else 0

    print(f"Tumor volume: {tumor_volume:.2f} mm³")
    print(f"Vessel density: {vessel_density:.4f}")
    
    return vessel_density
```

## 总结

### 特征计算方法总结

| 特征 | 计算方法 | 数学公式 | 物理意义 |
|------|---------|---------|---------|
| **曲率** | B样条拟合 + 导数计算 | κ = \|r' × r''\| / \|r'\|³ | 血管转弯角度 |
| **扭率** | 三阶导数计算 | τ = (r' × r'') · r''' / \|r' × r''\|² | 血管扭曲程度 |
| **分支密度** | 分叉点计数 / 肿瘤体积 | ρ = N_junctions / V_tumor | 血管分叉密度 |
| **血管密度** | 血管体积 / 肿瘤体积 | ρ = V_vessel / V_tumor | 血管丰富程度 |
| **半径** | 距离变换 | r = DT(x,y,z) | 血管粗细 |

### 技术要点

1. **分割引擎**：
   - 使用3D U-Net或nnU-Net进行自动分割
   - 多任务学习同时分割肿瘤和血管
   - 深度监督提高分割精度

2. **骨架化算法**：
   - 形态学细化：简单快速
   - 距离变换：保留半径信息
   - VTK方法：高质量中轴线

3. **特征计算**：
   - 曲率：反映血管弯曲程度
   - 扭率：反映血管扭曲程度
   - 分支密度：反映血管丰富程度

### 应用价值

这些特征在临床中具有重要价值：
- **肿瘤血管生成评估**：分支密度、血管密度
- **血管异常检测**：曲率、扭率异常
- **治疗反应预测**：血管形态变化
- **预后评估**：血管特征与生存期相关
